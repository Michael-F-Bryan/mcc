10/// An untyped reference to a [`tree_sitter::Node`] in a [`SourceFile`].
11#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash)]
12pub struct RawPtr {
13    pub file: SourceFile,
14    pub span: Span,
15    pub id: usize,
16}
17
18impl RawPtr {
19    pub fn for_node(file: SourceFile, node: tree_sitter::Node<'_>) -> Self {
20        let span = Span::for_node(node);
21
22        RawPtr {
23            file,
24            span,
25            id: node.id(),
26        }
27    }
28
29    pub fn node<'db>(&self, db: &'db dyn Db) -> tree_sitter::Node<'db> {
30        let tree = self.file.ast(db).tree(db);
31        let mut cursor = tree.walk();
32
33        loop {
34            let node = cursor.node();
35
36            if node.id() == self.id {
37                return node;
38            }
39
40            if cursor.goto_first_child_for_byte(self.span.start).is_none() {
41                panic!("failed to find node with id {}", self.id);
42            }
43        }
44    }
45}

47/// A strongly-typed pointer to a [`type_sitter::Node`] inside a [`SourceFile`].
48#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash)]
49#[repr(transparent)]
50pub struct Ptr<T> {
51    ptr: RawPtr,
52    _node_type: std::marker::PhantomData<T>,
53}
54
55impl<T: type_sitter::Node<'static>> Ptr<T> {
56    pub fn for_node<'tree>(file: SourceFile, node: T::WithLifetime<'tree>) -> Self {
57        let node = node.raw();
58        Ptr {
59            ptr: RawPtr::for_node(file, *node),
60            _node_type: std::marker::PhantomData,
61        }
62    }
63
64    pub fn node<'db>(self, db: &'db dyn Db) -> T::WithLifetime<'db> {
65        let node = self.ptr.node(db);
66        T::WithLifetime::<'db>::try_from_raw(node).unwrap()
67    }
68
69    pub fn raw(&self) -> RawPtr {
70        self.ptr
71    }
72}
73
74impl<'tree, T: type_sitter::Node<'tree>> Deref for Ptr<T> {
75    type Target = RawPtr;
76    fn deref(&self) -> &Self::Target {
77        &self.ptr
78    }
79}

81macro_rules! ptr_type {
82    (
83        $(
84            $(#[$meta:meta])*
85            pub struct $name:ident($node:ident);
86        )*
87    ) => {
88        $(
89            $(#[$meta])*
90            #[doc = concat!("A pointer to a [`mcc_syntax::ast::", stringify!($node), "`] node.")]
91            #[salsa::interned]
92            #[derive(Debug)]
93            pub struct $name<'db> {
94                pub ptr: $crate::ptr::Ptr<mcc_syntax::ast::$node<'static>>,
95            }
96
97            impl<'db> $name<'db> {
98                pub fn from_node(db: &'db dyn $crate::Db, source_file: $crate::types::SourceFile, node: mcc_syntax::ast::$node<'db>) -> Self {
99                    let ptr = $crate::ptr::Ptr::for_node(source_file, node);
100                    $name::new(db, ptr)
101                }
102
103                pub fn node(&self, db: &'db dyn $crate::Db) -> mcc_syntax::ast::$node<'db> {
104                    self.ptr(db).node(db)
105                }
106
107                pub fn file(&self, db: &'db dyn $crate::Db) -> $crate::types::SourceFile {
108                    self.ptr(db).file
109                }
110
111                pub fn span(&self, db: &'db dyn $crate::Db) -> mcc_syntax::Span {
112                    self.ptr(db).span
113                }
114            }
115        )*
116    };
117}

124#[cfg(test)]
125mod tests {
126    use tree_sitter::{Query, QueryCursor, StreamingIterator};
127
128    use super::*;
129
130    #[test]
131    fn round_trip_ptr() {
132        let db = salsa::DatabaseImpl::default();
133        let file = SourceFile::new(&db, "test.c".into(), "int main(void) {}".into());
134        let tree = file.ast(&db).tree(&db);
135
136        let lang = tree_sitter::Language::new(tree_sitter_c::LANGUAGE);
137        let query = Query::new(&lang, "(function_definition) @f").unwrap();
138        let mut cursor = QueryCursor::new();
139        let mut captures = cursor.captures(&query, tree.root_node(), file.contents(&db).as_bytes());
140        let (matches, idx) = captures.next().unwrap();
141        let node = matches.captures[*idx].node;
142
143        let ptr = dbg!(RawPtr::for_node(file, node));
144        let round_tripped = ptr.node(&db);
145        assert_eq!(round_tripped, node);
146    }
