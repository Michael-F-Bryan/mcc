23pub fn typecheck<'db>(db: &'db dyn Db, file: SourceFile) -> hir::TranslationUnit<'db> {
24    let ast = crate::parse(db, file);
25    if !crate::parse::accumulated::<Diagnostics>(db, file).is_empty() {
26        return hir::TranslationUnit::new(db, Vec::new(), file);
27    }
28
29    ensure_no_keyword_as_identifier(
30        db,
31        file,
32        ast.tree(db).root_node(),
33        file.contents(db).as_ref(),
34    );
35
36    let tu = ast.root(db);
37    let mut cursor = tu.walk();
38    let mut items = Vec::new();
39
40    for child in tu.children(&mut cursor).filter_map(|c| c.ok()) {
41        type Child<'db> = <ast::TranslationUnit<'db> as HasChildren<'db>>::Child;
42        match child {
43            Child::FunctionDefinition(f) => {
44                if let Ok(body) = f.body() {
45                    ensure_valid_declaration_types(db, file, body);
46                }
47                if let Some(hir_f) = function_definition(db, file, f) {
48                    let _ = function_signature(db, hir_f);
49                    items.push(hir::Item::Function(hir_f));
50                }
51            }
52            other => {
53                Diagnostic::bug()
54                    .with_message("Translation unit item not implemented")
55                    .with_code(codes::type_check::unimplemented)
56                    .with_labels(vec![
57                        Label::primary(file, Span::for_node(*other.raw()))
58                            .with_message(other.kind()),
59                    ])
60                    .accumulate(db);
61            }
62        }
63    }
64
65    hir::TranslationUnit::new(db, items, file)
66}

68/// Lazily compute a function's signature (return type and parameters) from the AST.
69/// Emits diagnostics for missing return type or invalid type specifier.
70#[salsa::tracked]
71fn function_signature<'db>(
72    db: &'db dyn Db,
73    f: hir::FunctionDefinition<'db>,
74) -> hir::Signature<'db> {
75    let node = f.node(db).node(db);
76    let file = f.node(db).file(db);
77    let src = file.contents(db);
78    let raw = node.raw();
79
80    let return_type = match raw.child_by_field_name("type") {
81        None => {
82            Diagnostic::error()
83                .with_message("Expected a return type for function")
84                .with_code(codes::type_check::missing_return_type)
85                .with_labels(vec![
86                    Label::primary(file, mcc_syntax::Span::for_node(*raw))
87                        .with_message("error occurred here"),
88                ])
89                .accumulate(db);
90            hir::Type::Error
91        }
92        Some(spec) => {
93            let kind = spec.kind();
94            let text = spec.utf8_text(src.as_bytes()).unwrap_or_default().trim();
95            let is_old_style = raw
96                .child_by_field_name("declarator")
97                .map_or(false, |d| d.kind() == "parenthesized_declarator");
98            if kind == "type_identifier" {
99                if is_old_style {
100                    Diagnostic::error()
101                        .with_message("Expected a return type for function")
102                        .with_code(codes::type_check::missing_return_type)
103                        .with_labels(vec![
104                            Label::primary(file, mcc_syntax::Span::for_node(spec))
105                                .with_message("error occurred here"),
106                        ])
107                        .accumulate(db);
108                } else {
109                    Diagnostic::error()
110                        .with_message(format!("invalid type \"{text}\""))
111                        .with_code(codes::type_check::invalid_type)
112                        .with_labels(vec![
113                            Label::primary(file, mcc_syntax::Span::for_node(spec))
114                                .with_message("invalid or unsupported type"),
115                        ])
116                        .accumulate(db);
117                }
118                hir::Type::Error
119            } else if is_old_style && text == "void" {
120                // Old-style main(void): type slot holds parameter's void, not return type.
121                Diagnostic::error()
122                    .with_message("Expected a return type for function")
123                    .with_code(codes::type_check::missing_return_type)
124                    .with_labels(vec![
125                        Label::primary(file, mcc_syntax::Span::for_node(spec))
126                            .with_message("error occurred here"),
127                    ])
128                    .accumulate(db);
129                hir::Type::Error
130            } else {
131                match text {
132                    "int" => hir::Type::Int,
133                    "void" => hir::Type::Void,
134                    _ => {
135                        Diagnostic::error()
136                            .with_message(format!("invalid type \"{text}\""))
137                            .with_code(codes::type_check::invalid_type)
138                            .with_labels(vec![
139                                Label::primary(file, mcc_syntax::Span::for_node(spec))
140                                    .with_message("invalid or unsupported type"),
141                            ])
142                            .accumulate(db);
143                        hir::Type::Error
144                    }
145                }
146            }
147        }
148    };
149
150    let parameters = Vec::new();
151
152    hir::Signature {
153        return_type,
154        parameters,
155    }
156}

158fn function_definition<'db>(
159    db: &'db dyn Db,
160    file: SourceFile,
161    f: ast::FunctionDefinition<'db>,
162) -> Option<hir::FunctionDefinition<'db>> {
163    let signature: ast::FunctionDeclarator<'db> = f.declarator().ok()?.as_function_declarator()?;
164    let ident: ast::Identifier<'db> = signature.declarator().ok()?.as_identifier()?;
165
166    let name = hir::Identifier::from_node(db, file, ident);
167    let ptr = ptr::FunctionDefinition::from_node(db, file, f);
168
169    Some(hir::FunctionDefinition::new(db, name, ptr))
170}

172/// C keywords may not be used as identifiers (e.g. `int return = 4;`).
173/// Only flag identifiers used as names (declarator or expression), not type names like `void` in `main(void)`.
174fn ensure_no_keyword_as_identifier(db: &dyn Db, file: SourceFile, root: TsNode, src: &str) {
175    let lang = Language::new(tree_sitter_c::LANGUAGE);
176    let query = tree_sitter::Query::new(&lang, "(identifier) @id").unwrap();
177    let mut cursor = tree_sitter::QueryCursor::new();
178    let mut matches = cursor.matches(&query, root, src.as_bytes());
179
180    while let Some(m) = matches.next() {
181        let node = m.captures[0].node;
182        if is_type_only_identifier(node) {
183            continue;
184        }
185        let text = node.utf8_text(src.as_bytes()).unwrap();
186        if C_KEYWORDS.contains(&text) {
187            Diagnostic::error()
188                .with_message(format!(
189                    "\"{text}\" is a keyword and cannot be used as an identifier"
190                ))
191                .with_code(codes::type_check::keyword_as_identifier)
192                .with_labels(vec![
193                    Label::primary(file, Span::for_node(node))
194                        .with_message("keyword used as identifier"),
195                ])
196                .accumulate(db);
197        }
198    }
199}

201/// Emit invalid-type diagnostic for declarations in a function body that use type_identifier (e.g. "ints a").
202fn ensure_valid_declaration_types<'db>(
203    db: &'db dyn Db,
204    file: SourceFile,
205    body: ast::CompoundStatement<'db>,
206) {
207    let src = file.contents(db);
208    let mut cursor = body.walk();
209    for child in body.children(&mut cursor).filter_map(|c| c.ok()) {
210        if let Some(decl) = child.as_declaration() {
211            let raw = decl.raw();
212            if let Some(type_spec) = raw.child_by_field_name("type") {
213                if type_spec.kind() == "type_identifier" {
214                    let text = type_spec
215                        .utf8_text(src.as_bytes())
216                        .unwrap_or_default()
217                        .trim();
218                    Diagnostic::error()
219                        .with_message(format!("invalid type \"{text}\""))
220                        .with_code(codes::type_check::invalid_type)
221                        .with_labels(vec![
222                            Label::primary(file, Span::for_node(type_spec))
223                                .with_message("invalid or unsupported type"),
224                        ])
225                        .accumulate(db);
226                }
227            }
228        }
229    }
230}

232/// True if this identifier is only used as a type (e.g. `void` in `main(void)`).
233fn is_type_only_identifier(node: TsNode) -> bool {
234    let mut cur = node;
235    while let Some(p) = cur.parent() {
236        if ast::Declarator::try_from_raw(p).is_ok()
237            || ast::InitDeclarator::try_from_raw(p).is_ok()
238            || ast::FunctionDeclarator::try_from_raw(p).is_ok()
239        {
240            return false;
241        }
242        if ast::TypeSpecifier::try_from_raw(p).is_ok()
243            || ast::ParameterDeclaration::try_from_raw(p).is_ok()
244            || ast::ParameterList::try_from_raw(p).is_ok()
245        {
246            return true;
247        }
248        cur = p;
249    }
250    false
251}

253/// A typechecking context - roughly equal to a "scope".
254#[salsa::tracked(debug)]
255pub struct TyCtx<'db> {
256    #[returns(ref)]
257    pub defs: Vec<hir::Def<'db>>,
258    pub parent: Option<TyCtx<'db>>,
259}
260
261#[salsa::tracked]
262impl<'db> TyCtx<'db> {
263    pub fn lookup(self, db: &'db dyn Db, name: hir::Identifier<'db>) -> Option<hir::Def<'db>> {
264        let names = self.names(db);
265        names.get(&name).copied()
266    }
267
268    pub fn names(&self, db: &'db dyn Db) -> im::OrdMap<hir::Identifier<'db>, hir::Def<'db>> {
269        let mut names = self.parent(db).map(|p| p.names(db)).unwrap_or_default();
270
271        for &def in self.defs(db) {
272            names.insert(def.name(db), def);
273        }
274
275        names
276    }
277}
