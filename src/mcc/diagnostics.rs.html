17/// A newtype wrapper around [`Diagnostic`] that is used to accumulate errors as
18/// the compiler runs.
19#[repr(transparent)]
20#[salsa::accumulator]
21#[derive(Debug, Clone, PartialEq, Eq)]
22pub struct Diagnostics(pub Diagnostic);
23
24impl From<Diagnostic> for Diagnostics {
25    fn from(diagnostic: Diagnostic) -> Self {
26        Diagnostics(diagnostic)
27    }

30/// Declarative macro for defining error codes in a hierarchical structure.
31///
32/// # Example
33/// ```rust
34/// mcc::codes! {
35///   parse {
36///     /// The parser encountered an unexpected token.
37///     const UNEXPECTED: &str = "unexpected";
38///   }
39///   types {
40///     /// This part of the type checker isn't implemented.
41///     const UNIMPLEMENTED: &str = "unimplemented";
42///   }
43/// }
44/// ```
45#[macro_export]
46macro_rules! codes {
47    // Base case: no more modules to process
48    () => {};
49
50    // Handle a module with constants
51    (
52        $(#[$mod_doc:meta])*
53        $module:ident {
54            $($(#[$doc:meta])* const $const_name:ident: &str = $value:expr;)*
55        }
56        $($rest:tt)*
57    ) => {
58        $( #[$mod_doc] )*
59        pub mod $module {
60            $(
61                $(#[$doc])*
62                pub const $const_name: &str = concat!(stringify!($module), "::", $value);
63            )*
64        }
65        $crate::codes!($($rest)*);
66    };
67
68    // Handle nested modules
69    (
70        $(#[$mod_doc:meta])*
71        $module:ident {
72            $($nested:tt)*
73        }
74        $($rest:tt)*
75    ) => {
76        $( #[$mod_doc] )*
77        pub mod $module {
78            $crate::codes!($($nested)*);
79        }
80        $crate::codes!($($rest)*);
81    };
82}
