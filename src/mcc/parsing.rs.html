11/// Parse a C program into an abstract syntax tree.
12#[tracing::instrument(level = "info", skip_all)]
13#[salsa::tracked]
14pub fn parse(db: &dyn Db, file: SourceFile) -> Ast<'_> {
15    let mut parser = tree_sitter::Parser::new();
16    let lang = Language::new(tree_sitter_c::LANGUAGE);
17    parser.set_language(&lang).unwrap();
18
19    let src = file.contents(db);
20    let tree = Tree::from(parser.parse(src, None).unwrap());
21
22    check_tree(db, &tree, file);
23    ensure_return_type(db, &lang, &tree, file);
24
25    Ast::new(db, tree)
26}

28/// the return type for a C function is treated as optional by the grammar, but
29/// we want it to be required.
30#[tracing::instrument(level = "debug", skip_all)]
31fn ensure_return_type(db: &dyn Db, lang: &Language, tree: &Tree, file: SourceFile) {
32    let query = tree_sitter::Query::new(
33        lang,
34        "(function_definition
35          type: (type_identifier) @missing-return-type
36          declarator: (parenthesized_declarator)) @function-def",
37    )
38    .unwrap();
39    let src = file.contents(db);
40
41    let mut cursor = tree_sitter::QueryCursor::new();
42    let mut captures = cursor.matches(&query, tree.root_node(), src.as_bytes());
43
44    while let Some(m) = captures.next() {
45        let diagnostic = codespan_reporting::diagnostic::Diagnostic::error()
46            .with_message("Expected a return type for function")
47            .with_code(codes::parse::missing_token)
48            .with_labels(vec![
49                Label::primary(file, Span::for_node(m.captures[0].node))
50                    .with_message("error occurred here"),
51            ]);
52        diagnostic.accumulate(db);
53    }
54}

56#[tracing::instrument(level = "debug", skip_all)]
57fn check_tree(db: &dyn Db, tree: &Tree, file: SourceFile) {
58    let mut cursor = tree.walk();
59
60    let mut to_check: Vec<Node<'_>> = vec![tree.root_node()];
61
62    while let Some(node) = to_check.pop() {
63        match check_node(db, node, file) {
64            Continuation::Skip => {}
65            Continuation::Recurse => {
66                cursor.reset(node);
67                for child in node.children(&mut cursor) {
68                    to_check.push(child);
69                }
70            }
71            Continuation::Emit(diag) => {
72                diag.accumulate(db);
73            }
74        }
75    }
76}

78fn check_node(db: &dyn Db, node: Node<'_>, file: SourceFile) -> Continuation {
79    if !node.has_error() {
80        Continuation::Skip
81    } else if node.is_missing() {
82        let diagnostic = Diagnostic::error()
83            .with_message(format!(
84                "Expected a \"{}\"",
85                node.parent().unwrap().grammar_name()
86            ))
87            .with_code(codes::parse::unexpected_token)
88            .with_labels(vec![
89                Label::primary(file, Span::for_node(node)).with_message("error occurred here"),
90            ]);
91        Continuation::Emit(diagnostic)
92    } else if node.is_error() {
93        let token = node.utf8_text(file.contents(db).as_ref()).unwrap();
94
95        let diagnostic = Diagnostic::error()
96            .with_message(format!(
97                "Expected a \"{}\", but found \"{}\"",
98                node.parent().unwrap().grammar_name(),
99                token
100            ))
101            .with_labels(vec![
102                Label::primary(file, Span::for_node(node)).with_message("error occurred here"),
103            ]);
104        Continuation::Emit(diagnostic)
105    } else {
106        Continuation::Recurse
107    }
108}
109
110#[derive(Debug, Clone)]
111enum Continuation {
112    /// Skip this node and all its children.
113    Skip,
114    /// Recurse into the children of this node.
115    Recurse,
116    /// Emit a diagnostic for this node.
117    Emit(Diagnostic),
118}
