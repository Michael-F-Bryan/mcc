9/// Run the C preprocessor on the given input.
10#[tracing::instrument(level = "info", skip_all)]
11#[salsa::tracked]
12pub fn preprocess(db: &dyn Db, cc: OsString, src: SourceFile) -> Result<Text, PreprocessorError> {
13    let path = src.path(db);
14
15    let mut cmd = Command::new(&cc);
16    cmd.arg("-E")
17        .arg("-P")
18        .arg(path.as_str())
19        .stdin(Stdio::null())
20        .stdout(Stdio::piped())
21        .stderr(Stdio::piped());
22
23    let Ok(output) = cmd.output() else {
24        let msg = format!(
25            "Failed to start \"{}\", is it installed?",
26            cc.to_string_lossy()
27        );
28        return Err(PreprocessorError {
29            cc,
30            path: PathBuf::from(path.as_str()),
31            message: msg.into(),
32        });
33    };
34
35    if !output.status.success() {
36        return Err(PreprocessorError {
37            cc,
38            path: PathBuf::from(path.as_str()),
39            message: String::from_utf8_lossy(&output.stderr).into(),
40        });
41    }
42
43    Ok(Text::from(String::from_utf8_lossy(&output.stdout)))
44}
45
46#[derive(Debug, Clone, PartialEq, Eq)]
47pub struct PreprocessorError {
48    pub cc: OsString,
49    pub path: PathBuf,
50    pub message: Text,
51}
52
53impl std::error::Error for PreprocessorError {}
54
55impl std::fmt::Display for PreprocessorError {
56    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
57        let PreprocessorError {
58            cc,
59            path: src,
60            message,
61        } = self;
62        let cc = cc.to_string_lossy();
63        let src = src.display();
64        let message = message.trim();
65        write!(f, "preprocessing \"{src}\"with \"{cc}\" failed: {message}")
66    }
67}
