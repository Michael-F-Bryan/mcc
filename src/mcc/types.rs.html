66/// A quick'n'dirty s-expression pretty-printer.
67fn format_sexpr(raw: &str) -> String {
68    let mut result = String::new();
69    let mut depth = 0;
70    let mut in_word = false;
71    let mut after_colon = false;
72    let mut chars = raw.chars().peekable();
73
74    while let Some(c) = chars.next() {
75        match c {
76            '(' => {
77                if in_word {
78                    result.push(' ');
79                    in_word = false;
80                }
81                if !after_colon {
82                    result.push('\n');
83                    result.extend(std::iter::repeat_n("  ", depth));
84                }
85                result.push('(');
86                depth += 1;
87                after_colon = false;
88            }
89            ')' => {
90                depth = depth.checked_sub(1).expect("Mismatched parentheses");
91                result.push(')');
92                in_word = false;
93                after_colon = false;
94            }
95            ' ' | '\n' | '\t' => {
96                if in_word {
97                    result.push(' ');
98                    in_word = false;
99                }
100                // Preserve original whitespace after fields
101                if after_colon {
102                    result.push(c);
103                }
104            }
105            ':' => {
106                result.push(c);
107                after_colon = true;
108                in_word = true;
109            }
110            '\\' => {
111                // Handle escaped characters
112                if let Some(next_char) = chars.next() {
113                    // For escaped quotes, just output the quote without the backslash
114                    if next_char == '"' {
115                        result.push('"');
116                    } else {
117                        result.push(c);
118                        result.push(next_char);
119                    }
120                    in_word = true;
121                } else {
122                    result.push(c);
123                    in_word = true;
124                }
125            }
126            _ => {
127                // Only add newline if we're not after a colon and not already at the start of a line
128                if !in_word && !result.ends_with('(') && !after_colon {
129                    result.push('\n');
130                    result.extend(std::iter::repeat_n("  ", depth));
131                }
132                result.push(c);
133                in_word = true;
134                // Reset after_colon only for non-whitespace characters
135                if c != ' ' && c != '\n' && c != '\t' {
136                    after_colon = false;
137                }
138            }
139        }
140    }
141
142    let trimmed = result.trim_start().to_string();
143
144    // Only add leading newline for complex nested structures
145    // Check if the input has multiple nested levels or is the translation_unit case
146    if raw.contains("translation_unit") || raw.matches('(').count() > 3 {
147        format!("\n{trimmed}")
148    } else {
149        trimmed
150    }
151}
