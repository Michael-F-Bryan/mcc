7/// Compile a parsed C program into assembly.
8#[tracing::instrument(level = "info", skip_all)]
9#[salsa::tracked]
10pub fn generate_assembly<'db>(db: &'db dyn Db, program: tacky::Program<'db>) -> asm::Program<'db> {
11    let mut functions = Vec::new();
12    for function in program.functions(db) {
13        functions.push(lower_function(db, function));
14    }
15
16    asm::Program::new(db, functions)
17}

19/// Lower from [Three Address Code](crate::lowering::tacky) to [Assembly](asm).
20#[tracing::instrument(level = "debug", skip_all, fields(name = %function.name(db)))]
21#[salsa::tracked]
22fn lower_function<'db>(
23    db: &'db dyn Db,
24    function: tacky::FunctionDefinition<'db>,
25) -> asm::FunctionDefinition<'db> {
26    let asm = to_assembly(db, function);
27    fix_up_instructions(db, asm)
28}

30/// Lower a [`tacky::FunctionDefinition`] to [`asm::FunctionDefinition`],
31/// placing variables and pseudo-registers on the stack.
32#[tracing::instrument(level = "debug", skip_all)]
33#[salsa::tracked]
34fn to_assembly<'db>(
35    db: &'db dyn Db,
36    function: tacky::FunctionDefinition<'db>,
37) -> asm::FunctionDefinition<'db> {
38    let name = function.name(db);
39    let mut instructions = Vec::new();
40    let mut stack_locations = StackAllocator::default();
41
42    for instruction in function.instructions(db) {
43        match instruction {
44            tacky::Instruction::Return(ret) => {
45                let src = stack_locations.operand_for(ret);
46
47                instructions.push(asm::Instruction::Mov {
48                    src,
49                    dst: asm::Operand::Register(asm::Register::AX),
50                });
51                instructions.push(asm::Instruction::Ret);
52            }
53            tacky::Instruction::Unary { op, src, dst } => {
54                let op = unary_operator_to_asm(op);
55                let src = stack_locations.operand_for(src);
56                let dst = stack_locations.operand_for(dst);
57
58                instructions.push(asm::Instruction::Mov { src, dst });
59                instructions.push(asm::Instruction::Unary { op, operand: dst });
60            }
61            tacky::Instruction::Binary {
62                op,
63                left_src,
64                right_src,
65                dst,
66            } => {
67                let left_src = stack_locations.operand_for(left_src);
68                let right_src = stack_locations.operand_for(right_src);
69                let dst = stack_locations.operand_for(dst);
70
71                enum BinOpKind {
72                    Bin(asm::BinaryOperator),
73                    Div,
74                    Mod,
75                }
76
77                let op = match op {
78                    tacky::BinaryOperator::Add => BinOpKind::Bin(asm::BinaryOperator::Add),
79                    tacky::BinaryOperator::Sub => BinOpKind::Bin(asm::BinaryOperator::Sub),
80                    tacky::BinaryOperator::Mul => BinOpKind::Bin(asm::BinaryOperator::Mul),
81                    tacky::BinaryOperator::And => BinOpKind::Bin(asm::BinaryOperator::And),
82                    tacky::BinaryOperator::Or => BinOpKind::Bin(asm::BinaryOperator::Or),
83                    tacky::BinaryOperator::LeftShift => {
84                        BinOpKind::Bin(asm::BinaryOperator::LeftShift)
85                    }
86                    tacky::BinaryOperator::RightShift => {
87                        BinOpKind::Bin(asm::BinaryOperator::RightShift)
88                    }
89                    tacky::BinaryOperator::Div => BinOpKind::Div,
90                    tacky::BinaryOperator::Mod => BinOpKind::Mod,
91                };
92
93                match op {
94                    BinOpKind::Bin(op) => {
95                        instructions.push(asm::Instruction::Mov {
96                            src: left_src,
97                            dst: asm::Operand::Register(asm::Register::R10),
98                        });
99                        instructions.push(asm::Instruction::Binary {
100                            op,
101                            src: right_src,
102                            dst: asm::Operand::Register(asm::Register::R10),
103                        });
104                        instructions.push(asm::Instruction::Mov {
105                            src: asm::Operand::Register(asm::Register::R10),
106                            dst,
107                        });
108                    }
109                    BinOpKind::Div => {
110                        instructions.push(asm::Instruction::Mov {
111                            src: left_src,
112                            dst: asm::Operand::Register(asm::Register::AX),
113                        });
114                        instructions.push(asm::Instruction::Cdq);
115                        instructions.push(asm::Instruction::Idiv { src: right_src });
116                        instructions.push(asm::Instruction::Mov {
117                            src: asm::Operand::Register(asm::Register::AX),
118                            dst,
119                        });
120                    }
121                    BinOpKind::Mod => {
122                        instructions.push(asm::Instruction::Mov {
123                            src: left_src,
124                            dst: asm::Operand::Register(asm::Register::AX),
125                        });
126                        instructions.push(asm::Instruction::Cdq);
127                        instructions.push(asm::Instruction::Idiv { src: right_src });
128                        instructions.push(asm::Instruction::Mov {
129                            src: asm::Operand::Register(asm::Register::DX),
130                            dst,
131                        });
132                    }
133                }
134            }
135            tacky::Instruction::Comparison {
136                op,
137                left_src,
138                right_src,
139                dst,
140            } => {
141                let left_src = stack_locations.operand_for(left_src);
142                let right_src = stack_locations.operand_for(right_src);
143                let dst = stack_locations.operand_for(dst);
144
145                let comparison_op = match op {
146                    tacky::ComparisonOperator::Equal => asm::ComparisonOperator::Equal,
147                    tacky::ComparisonOperator::NotEqual => asm::ComparisonOperator::NotEqual,
148                    tacky::ComparisonOperator::LessThan => asm::ComparisonOperator::LessThan,
149                    tacky::ComparisonOperator::LessThanOrEqual => {
150                        asm::ComparisonOperator::LessThanOrEqual
151                    }
152                    tacky::ComparisonOperator::GreaterThan => asm::ComparisonOperator::GreaterThan,
153                    tacky::ComparisonOperator::GreaterThanOrEqual => {
154                        asm::ComparisonOperator::GreaterThanOrEqual
155                    }
156                };
157
158                instructions.push(asm::Instruction::Comparison {
159                    op: comparison_op,
160                    left: left_src,
161                    right: right_src,
162                    dst,
163                });
164            }
165            tacky::Instruction::Copy { src, dst } => {
166                let src = stack_locations.operand_for(src);
167                let dst = stack_locations.operand_for(dst);
168
169                instructions.push(asm::Instruction::Mov { src, dst });
170            }
171            tacky::Instruction::Jump { target } => {
172                instructions.push(asm::Instruction::Jump { target });
173            }
174            tacky::Instruction::Label(target) => {
175                instructions.push(asm::Instruction::Label(target));
176            }
177            tacky::Instruction::JumpIfZero { condition, target } => {
178                let condition = stack_locations.operand_for(condition);
179                instructions.push(asm::Instruction::JumpIfZero { condition, target });
180            }
181            tacky::Instruction::JumpIfNotZero { condition, target } => {
182                let condition = stack_locations.operand_for(condition);
183                instructions.push(asm::Instruction::JumpIfNotZero { condition, target });
184            }
185        }
186    }
187
188    // Allocate stack space for local variables if needed. Each slot is 4 bytes.
189    let stack_size_bytes = (stack_locations.variables.len() as u32) * 4;
190    if stack_size_bytes > 0 {
191        instructions.insert(0, asm::Instruction::AllocateStack(stack_size_bytes));
192    }
193
194    asm::FunctionDefinition::new(db, name, instructions, function.span(db))
195}

197fn unary_operator_to_asm(op: tacky::UnaryOperator) -> asm::UnaryOperator {
198    match op {
199        tacky::UnaryOperator::Negate => asm::UnaryOperator::Neg,
200        tacky::UnaryOperator::Complement => asm::UnaryOperator::Complement,
201        tacky::UnaryOperator::Not => asm::UnaryOperator::Not,
202    }
203}

205#[derive(Debug, Default, Clone, PartialEq)]
206struct StackAllocator {
207    variables: Vec<tacky::Variable>,
208}
209
210impl StackAllocator {
211    fn operand_for(&mut self, val: tacky::Val) -> asm::Operand {
212        match val {
213            tacky::Val::Constant(c) => asm::Operand::Imm(c),
214            tacky::Val::Var(v) => asm::Operand::Stack(self.offset_for(v)),
215        }
216    }
217
218    fn offset_for(&mut self, variable: tacky::Variable) -> u32 {
219        (self.index_of(variable) as u32) * 4
220    }
221
222    fn index_of(&mut self, variable: tacky::Variable) -> usize {
223        match self.variables.iter().position(|v| v == &variable) {
224            Some(i) => i,
225            None => {
226                let index = self.variables.len();
227                self.variables.push(variable);
228                index
229            }
230        }
231    }
232}

234/// Fixes up invalid `mov` instructions where both source and destination are
235/// stack operands.
236///
237/// When replacing pseudoregisters with stack addresses, we may end up with
238/// `mov` instructions that have memory addresses as both source and destination
239/// operands. This is invalid assembly, as instructions like `movl -4(%rbp),
240/// -8(%rbp)` will be rejected by the assembler.
241///
242/// The fix is to rewrite such instructions to:
243/// 1. First copy from the source stack location into R10D
244/// 2. Then copy from R10D to the destination stack location
245#[tracing::instrument(level = "debug", skip_all)]
246#[salsa::tracked]
247fn fix_up_instructions<'db>(
248    db: &'db dyn Db,
249    function: asm::FunctionDefinition<'db>,
250) -> asm::FunctionDefinition<'db> {
251    let mut instructions = Vec::new();
252
253    for instruction in function.instructions(db) {
254        match instruction {
255            asm::Instruction::Mov {
256                src: src @ asm::Operand::Stack(_),
257                dst: dst @ asm::Operand::Stack(_),
258            } => {
259                // `mov` instructions with memory addresses as both source and
260                // destination are invalid assembly, so we need to move the
261                // source to a register first.
262                instructions.push(asm::Instruction::Mov {
263                    src,
264                    dst: asm::Operand::Register(asm::Register::R10),
265                });
266                instructions.push(asm::Instruction::Mov {
267                    src: asm::Operand::Register(asm::Register::R10),
268                    dst,
269                });
270            }
271            asm::Instruction::Idiv {
272                src: src @ asm::Operand::Imm(_),
273            } => {
274                // `idiv` does not accept an immediate, so we need to move the
275                // source to a register first.
276                instructions.push(asm::Instruction::Mov {
277                    src,
278                    dst: asm::Operand::Register(asm::Register::R10),
279                });
280                instructions.push(asm::Instruction::Idiv {
281                    src: asm::Operand::Register(asm::Register::R10),
282                });
283            }
284            asm::Instruction::Comparison {
285                op,
286                left: left @ asm::Operand::Imm(_),
287                right: right @ asm::Operand::Imm(_),
288                dst,
289            } => {
290                // `cmpl` does not accept two immediates, so we need to move one
291                // to a register first.
292                instructions.push(asm::Instruction::Mov {
293                    src: left,
294                    dst: asm::Operand::Register(asm::Register::R10),
295                });
296                instructions.push(asm::Instruction::Comparison {
297                    op,
298                    left: asm::Operand::Register(asm::Register::R10),
299                    right,
300                    dst,
301                });
302            }
303            asm::Instruction::Comparison {
304                op,
305                left: left @ asm::Operand::Stack(_),
306                right: right @ asm::Operand::Imm(_),
307                dst,
308            } => {
309                // `cmpl` does not accept memory as destination with immediate source,
310                // so we need to move the memory operand to a register first.
311                instructions.push(asm::Instruction::Mov {
312                    src: left,
313                    dst: asm::Operand::Register(asm::Register::R10),
314                });
315                instructions.push(asm::Instruction::Comparison {
316                    op,
317                    left: asm::Operand::Register(asm::Register::R10),
318                    right,
319                    dst,
320                });
321            }
322            asm::Instruction::Comparison {
323                op,
324                left: left @ asm::Operand::Imm(_),
325                right: right @ asm::Operand::Stack(_),
326                dst,
327            } => {
328                // `cmpl` does not accept memory as destination with immediate source,
329                // so we need to move the memory operand to a register first.
330                instructions.push(asm::Instruction::Mov {
331                    src: right,
332                    dst: asm::Operand::Register(asm::Register::R10),
333                });
334                instructions.push(asm::Instruction::Comparison {
335                    op,
336                    left: asm::Operand::Register(asm::Register::R10),
337                    right: left,
338                    dst,
339                });
340            }
341            other => instructions.push(other),
342        }
343    }
344
345    asm::FunctionDefinition::new(db, function.name(db), instructions, function.span(db))
346}
