1use std::process::{Command, Output};
2
3use crate::Text;
4
5/// Run a command and return its output.
6///
7/// If the command fails, this function will return an error.
8#[tracing::instrument(level = "debug")]
9pub(crate) fn run_cmd(cmd: &mut Command) -> Result<Output, CommandError> {
10    let output = cmd.output().map_err(|e| CommandError::StartFailed {
11        cmd: format!("{cmd:?}").into(),
12        error: e.to_string().into(),
13    })?;
14
15    if !output.status.success() {
16        return Err(CommandError::CompletedUnsuccessfully {
17            cmd: format!("{cmd:?}").into(),
18            status: output.status.code().unwrap_or(1),
19            stderr: String::from_utf8_lossy(&output.stderr).into(),
20        });
21    }
22
23    Ok(output)
24}
25
26#[derive(Debug, Clone, PartialEq, Eq, Hash, thiserror::Error)]
27pub enum CommandError {
28    #[error("unable to start \"{cmd}\": {error:?}")]
29    StartFailed { cmd: Text, error: Text },
30    #[error("command \"{cmd}\" completed unsuccessfully: {status}\n{stderr}")]
31    CompletedUnsuccessfully {
32        cmd: Text,
33        status: i32,
34        stderr: Text,
35    },
36}
