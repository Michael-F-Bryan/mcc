10/// Render a set of assembly instructions as a string.
11#[tracing::instrument(level = "debug", skip_all, fields(target = %target))]
12#[salsa::tracked]
13pub fn render_program<'db>(
14    db: &'db dyn Db,
15    program: asm::Program<'db>,
16    target: Triple,
17) -> Result<Text, fmt::Error> {
18    let mut output = String::new();
19    let mut renderer = AssemblyRenderer::new(target, &mut output);
20    renderer.program(db, program)?;
21    Ok(output.into())
22}
23
24#[derive(Debug, Clone, PartialEq, Eq)]
25struct AssemblyRenderer<W> {
26    target: Triple,
27    writer: W,
28}
29
30impl<W: Write> AssemblyRenderer<W> {
31    fn new(target: Triple, writer: W) -> Self {
32        Self { target, writer }
33    }
34
35    fn program(&mut self, db: &dyn Db, program: asm::Program) -> fmt::Result {
36        for function in program.functions(db) {
37            self.render_function(db, function)?;
38            writeln!(self.writer)?;
39        }
40
41        if self.target.operating_system == OperatingSystem::Linux {
42            writeln!(self.writer, ".section .note.GNU-stack, \"\", @progbits")?;
43        }
44
45        Ok(())
46    }
47
48    fn function_name<'a>(&self, name: &'a str) -> Cow<'a, str> {
49        if matches!(
50            self.target.operating_system,
51            OperatingSystem::MacOSX(_) | OperatingSystem::Darwin(_)
52        ) {
53            format!("_{name}").into()
54        } else {
55            name.into()
56        }
57    }
58
59    pub fn render_function(
60        &mut self,
61        db: &dyn Db,
62        function: asm::FunctionDefinition,
63    ) -> fmt::Result {
64        let name = function.name(db);
65        let name = self.function_name(&name);
66
67        writeln!(self.writer, ".globl {name}")?;
68        writeln!(self.writer, "{name}:")?;
69        writeln!(self.writer, "pushq %rbp")?;
70        writeln!(self.writer, "movq %rsp, %rbp")?;
71
72        for instruction in function.instructions(db) {
73            write!(self.writer, "  ")?;
74            self.render_instruction(instruction)?;
75        }
76
77        Ok(())
78    }
79
80    fn render_instruction(&mut self, instruction: asm::Instruction) -> fmt::Result {
81        match instruction {
82            asm::Instruction::AllocateStack(size) => {
83                writeln!(self.writer, "subq ${size}, %rsp")?;
84            }
85            asm::Instruction::Mov { src, dst } => {
86                write!(self.writer, "movl ")?;
87                self.operand(src)?;
88                write!(self.writer, ", ")?;
89                self.operand(dst)?;
90                writeln!(self.writer)?;
91            }
92            asm::Instruction::Unary { op, operand } => {
93                self.unary_operator(op)?;
94                write!(self.writer, " ")?;
95                self.operand(operand)?;
96                writeln!(self.writer)?;
97            }
98            asm::Instruction::Ret => {
99                writeln!(self.writer, "movq %rbp, %rsp")?;
100                writeln!(self.writer, "popq %rbp")?;
101                writeln!(self.writer, "ret")?;
102            }
103            asm::Instruction::Binary { op, src, dst } => {
104                self.binary_operator(op)?;
105                write!(self.writer, " ")?;
106                self.operand(src)?;
107                write!(self.writer, ", ")?;
108                self.operand(dst)?;
109                writeln!(self.writer)?;
110            }
111            asm::Instruction::Idiv { src } => {
112                write!(self.writer, "idivl ")?;
113                self.operand(src)?;
114                writeln!(self.writer)?;
115            }
116            asm::Instruction::Cdq => {
117                writeln!(self.writer, "cdq")?;
118            }
119        }
120
121        Ok(())
122    }
123
124    fn operand(&mut self, operand: asm::Operand) -> fmt::Result {
125        match operand {
126            asm::Operand::Imm(imm) => write!(self.writer, "${imm}"),
127            asm::Operand::Register(reg) => self.register(reg),
128            asm::Operand::Stack(stack) => write!(self.writer, "-{}(%rbp)", stack + 4),
129        }
130    }
131
132    fn register(&mut self, reg: asm::Register) -> fmt::Result {
133        match reg {
134            asm::Register::AX => write!(self.writer, "%eax"),
135            asm::Register::DX => write!(self.writer, "%edx"),
136            asm::Register::R10 => write!(self.writer, "%r10d"),
137        }
138    }
139
140    fn unary_operator(&mut self, op: asm::UnaryOperator) -> fmt::Result {
141        match op {
142            asm::UnaryOperator::Neg => write!(self.writer, "negl"),
143            asm::UnaryOperator::Not => write!(self.writer, "notl"),
144        }
145    }
146
147    fn binary_operator(&mut self, op: asm::BinaryOperator) -> fmt::Result {
148        match op {
149            asm::BinaryOperator::Add => write!(self.writer, "addl"),
150            asm::BinaryOperator::Sub => write!(self.writer, "subl"),
151            asm::BinaryOperator::Mul => write!(self.writer, "imull"),
152            asm::BinaryOperator::And => write!(self.writer, "andl"),
153            asm::BinaryOperator::Or => write!(self.writer, "orl"),
154            asm::BinaryOperator::LeftShift => write!(self.writer, "shll"),
155            asm::BinaryOperator::RightShift => write!(self.writer, "shrl"),
156        }
157    }
158}
