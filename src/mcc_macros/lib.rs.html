2use syn::{Data, DeriveInput, parse_macro_input, spanned::Spanned};

4#[proc_macro_derive(SerializeWithDatabase)]
5pub fn serialize_with_database(input: proc_macro::TokenStream) -> proc_macro::TokenStream {
6    let input = parse_macro_input!(input as DeriveInput);
7
8    let tokens = match &input.data {
9        Data::Struct(syn::DataStruct { fields: syn::Fields::Named(fields), .. }) => {
10             serialize_named_fields(&input, &fields.named)
11        }
12        _ => syn::Error::new(input.span(), "SerializeWithDatabase can only be used on structs with named fields. Did you put the #[derive(SerializeWithDatabase)] above #[salsa::tracked]?")
13            .into_compile_error(),
14    };
15
16    tokens.into()
17}
18
19fn serialize_named_fields(
20    input: &DeriveInput,
21    fields: &syn::punctuated::Punctuated<syn::Field, syn::token::Comma>,
22) -> proc_macro2::TokenStream {
23    let (parent_impl_generics, parent_ty_generics, parent_where_clause) =
24        input.generics.split_for_impl();
25
26    let ident = &input.ident;
27    let name = ident.to_string();
28
29    let serialize_fields = fields.iter().map(|field| {
30        let field_name = field.ident.as_ref().unwrap();
31        let as_string = field_name.to_string();
32        quote! {
33            {
34                let value = self.inner.#field_name(self.db);
35                let helper = crate::debug::helper(&value);
36                let ser = helper.serialize_with_db(self.db);
37                state.serialize_field(#as_string, &ser)?;
38            }
39        }
40    });
41    let num_fields = fields.len();
42
43    let mut child_generics = input.generics.clone();
44    let lifetime: syn::GenericParam = syn::parse_quote!('_ref);
45    child_generics.params.push(lifetime.clone());
46    let (child_impl_generics, child_ty_generics, child_where_clause) =
47        child_generics.split_for_impl();
48
49    quote! {
50        const _: () = {
51            use serde::ser::SerializeStruct;
52            use crate::debug::SerializeWithDatabase;
53
54            impl #parent_impl_generics SerializeWithDatabase for #ident #parent_ty_generics #parent_where_clause {
55                fn serialize_with_db<'a>(&'a self, db: &'a dyn salsa::Database) -> impl serde::Serialize + 'a {
56                    struct Impl #child_ty_generics #child_where_clause {
57                        db: & #lifetime dyn salsa::Database,
58                        inner: & #lifetime #ident #parent_ty_generics,
59                    }
60                    impl #child_impl_generics serde::Serialize for Impl #child_ty_generics #child_where_clause {
61                        fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
62                        where
63                            S: serde::Serializer,
64                        {
65                            let mut state = serializer.serialize_struct(#name, #num_fields)?;
66                            #(#serialize_fields)*
67                            state.end()
68                        }
69                    }
70                    Impl { db, inner: self }
71                }
72            }
73        };
74    }
75}
