1//! `escape8259` performs RFC8259-compliant string escaping and un-escaping.
2//!
3//! [RFC8259] is a JSON encoding standard.  Many JSON encoders exist, but other
4//! RFCs use the same string escaping mechanism, so it's useful to be able to
5//! access the string escaping functions by themselves.
6//!
7//! # Examples
8//!
9//! ```
10//! use escape8259::{escape, unescape};
11//!
12//! let s = "A null (\0) and a double-quote (\")";
13//! assert_eq!(escape(s), r#"A null (\u0000) and a double-quote (\")"#);
14//!
15//! let crab = r#"This is a crab: \ud83e\udd80"#;
16//! assert_eq!(unescape(crab).unwrap(), "This is a crab: ü¶Ä");
17//!
18//! // We accept encodings that weren't really necessary.
19//! assert_eq!(unescape(r#"\u0041\n"#).unwrap(), "A\n");
20//!
21//! let multiline = r#"hello
22//!  world"#;
23//! assert_eq!(escape(multiline), r#"hello\n world"#);
24//! ```
25//!
26//! [RFC8259]: https://tools.ietf.org/html/rfc8259
27

35/// An error occurred while unescaping.
36#[allow(clippy::empty_structs_with_brackets)] // FIXME: correct this if releasing a new major version.
37#[derive(Debug, Clone, Eq, PartialEq)]
38pub struct UnescapeError {}
39
40impl Display for UnescapeError {
41    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
42        f.write_str("failed rfc8259 unescape")
43    }
44}
45
46impl std::error::Error for UnescapeError {}

48type UnescapeResult<T> = Result<T, UnescapeError>;
49
50// Used to collect output characters and queue u16 values for translation.
51struct UnescapeState {
52    // The accumulated characters
53    out: String,
54    // Store a fragment of a large character for later decoding
55    stash: u16,
56}
57
58impl UnescapeState {
59    fn new() -> UnescapeState {
60        UnescapeState {
61            out: String::new(),
62            stash: 0,
63        }
64    }
65
66    // Collect a new character
67    fn push_char(&mut self, c: char) -> UnescapeResult<()> {
68        if self.stash != 0 {
69            return Err(UnescapeError {});
70        }
71        self.out.push(c);
72        Ok(())
73    }
74
75    // Collect a new UTF16 word.  This can either be one whole character,
76    // or part of a larger character.
77    fn push_u16(&mut self, x: u16) -> UnescapeResult<()> {
78        let surrogate = (0xD800..=0xDFFF).contains(&x);
79        match (self.stash, surrogate) {
80            (0, false) => {
81                // The std library only provides utf16 decode of an iterator,
82                // so to decode a single character we wrap it in an array.
83                // Hopefully the compiler will elide most of this extra work.
84                let words = [x];
85                match decode_utf16(words.iter().copied()).next() {
86                    Some(Ok(c)) => {
87                        self.out.push(c);
88                    }
89                    _ => return Err(UnescapeError {}),
90                }
91            }
92            (0, true) => self.stash = x,
93            (_, false) => {
94                return Err(UnescapeError {});
95            }
96            (w, true) => {
97                let words = [w, x];
98                match decode_utf16(words.iter().copied()).next() {
99                    Some(Ok(c)) => {
100                        self.out.push(c);
101                        self.stash = 0;
102                    }
103                    _ => return Err(UnescapeError {}),
104                }
105            }
106        }
107        Ok(())
108    }
109
110    // If we queued up part of a UTF-16 encoded word but didn't
111    // finish it, return an error.  Otherwise, consume self and
112    // return the accumulated String.
113    fn finalize(self) -> UnescapeResult<String> {
114        if self.stash != 0 {
115            return Err(UnescapeError {});
116        }
117        Ok(self.out)
118    }
119}

121fn parse_u16<S>(s: &mut S) -> UnescapeResult<u16>
122where
123    S: Iterator<Item = char>,
124{
125    // Placeholder character in case the input doesn't have the 4 chars we want.
126    let placeholders = std::iter::repeat('\0');
127    let hexnum: String = s.chain(placeholders).take(4).collect();
128    u16::from_str_radix(&hexnum, 16).map_err(|_| UnescapeError {})
129}

131// RFC8259 says non-escaped characters must be in one of the following ranges:
132// %x20-21 / %x23-5B / %x5D-10FFFF
133fn is_safe_char(c: char) -> bool {
134    let safe_ranges = [(0x20..=0x21), (0x23..=0x5B), (0x5D..=0x10FFFF)];
135    let cv = c as u32;
136
137    safe_ranges.iter().any(|range| range.contains(&cv))
138}

140/// Un-escape a string, following RFC8259 rules.
141///
142/// The only allowed single-character escapes are:
143/// `\" \\ \/ /b /f /n /r /t`
144///
145/// Any other character may be escaped in UTF-16 form:
146/// `\uXXXX` or `\uXXXX\uXXXX`
147///
148/// Characters in the ranges `0x20-21`, `0x23-5B`, `0x5D-10FFFF`
149/// may appear un-escaped in the input.
150#[inline]
151pub fn unescape<S>(s: S) -> UnescapeResult<String>
152where
153    S: AsRef<str>,
154{
155    unescape_inner(s.as_ref())
156}
157
158fn unescape_inner(s: &str) -> UnescapeResult<String> {
159    let mut state = UnescapeState::new();
160    let mut ins = s.chars();
161
162    while let Some(c) = ins.next() {
163        if c == '\\' {
164            match ins.next() {
165                None => {
166                    return Err(UnescapeError {});
167                }
168                Some(d) => {
169                    match d {
170                        '"' | '\\' | '/' => state.push_char(d)?,
171                        'b' => state.push_char('\x08')?, // backspace
172                        'f' => state.push_char('\x0C')?, // formfeed
173                        'n' => state.push_char('\n')?,   // linefeed
174                        'r' => state.push_char('\r')?,   // carriage return
175                        't' => state.push_char('\t')?,   // tab
176                        'u' => {
177                            let val = parse_u16(&mut ins)?;
178                            state.push_u16(val)?;
179                        }
180                        _ => {
181                            return Err(UnescapeError {});
182                        }
183                    }
184                }
185            }
186        } else if is_safe_char(c) {
187            state.push_char(c)?;
188        } else {
189            return Err(UnescapeError {});
190        }
191    }
192
193    state.finalize()
194}

196// %x22 /          ; "    quotation mark  U+0022
197// %x5C /          ; \    reverse solidus U+005C
198// %x2F /          ; /    solidus         U+002F
199// %x62 /          ; b    backspace       U+0008
200// %x66 /          ; f    form feed       U+000C
201// %x6E /          ; n    line feed       U+000A
202// %x72 /          ; r    carriage return U+000D
203// %x74 /          ; t    tab             U+0009
204// %x75 4HEXDIG )  ; uXXXX                U+XXXX
205
206fn force_escape(c: char, out: &mut String) {
207    let c = c as u32;
208    match c {
209        0x08 => out.push_str("\\b"),
210        0x09 => out.push_str("\\t"),
211        0x0A => out.push_str("\\n"),
212        0x0C => out.push_str("\\f"),
213        0x0D => out.push_str("\\r"),
214        0x22 => out.push_str("\\\""),
215        0x5C => out.push_str("\\\\"),
216        _ => {
217            // RFC8259 allows unicode characters natively, so there is no need
218            // to convert everything into \uXXXX form.  The only thing that's
219            // required to use that form are the ASCII control characters,
220            // which will never require more than one \uXXXX value.
221            if c >= 0x20 {
222                panic!("force_escape unnecessary encoding requested");
223            }
224            write!(out, "\\u{:04x}", c).unwrap();
225        }
226    }
227}

229/// Escape a string, following RFC8259 rules.
230///
231/// Only characters that require escaping will be escaped:
232/// quotation mark `?`,
233/// reverse solidus `\` (backslash),
234/// and the control characters (`0x00-1F`).
235#[inline]
236pub fn escape<S>(s: S) -> String
237where
238    S: AsRef<str>,
239{
240    escape_inner(s.as_ref())
241}
242
243fn escape_inner(s: &str) -> String {
244    let mut out = String::new();
245    for c in s.chars() {
246        if is_safe_char(c) {
247            out.push(c);
248        } else {
249            force_escape(c, &mut out);
250        }
251    }
252    out
253}
