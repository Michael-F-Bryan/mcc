14/// A lock-free, append-only vector.
15pub struct Vec<T> {
16    /// A counter used to retrieve a unique index to push to.
17    ///
18    /// This value may be more than the true length as it will
19    /// be incremented before values are actually stored.
20    inflight: AtomicUsize,
21
22    /// Buckets of length 32, 64 .. 2^62.
23    buckets: [Bucket<T>; BUCKETS],
24
25    /// The number of initialized elements in this vector.
26    count: AtomicUsize,
27}
28
29/// Safety: A `Vec` is owned and owns its elements, so sending a
30/// vector only sends its elements, hence `T: Send`.
31unsafe impl<T: Send> Send for Vec<T> {}
32
33/// Safety: Sharing a `Vec` exposes shared access to the
34/// elements inside, hence `T: Sync`. Additionally, a `Vec`
35/// may act as a channel, exposing owned access of elements
36/// to other threads, hence we also require `T: Send`.
37unsafe impl<T: Send + Sync> Sync for Vec<T> {}
38
39impl<T> Vec<T> {
40    /// An empty vector.
41    #[cfg(not(loom))]
42    const EMPTY: Vec<T> = Vec {
43        inflight: AtomicUsize::new(0),
44        buckets: [Bucket::EMPTY; BUCKETS],
45        count: AtomicUsize::new(0),
46    };
47
48    /// Create an empty vector.
49    #[cfg(not(loom))]
50    pub const fn new() -> Vec<T> {
51        Vec::EMPTY
52    }
53
54    /// Create an empty vector.
55    #[cfg(loom)]
56    pub fn new() -> Vec<T> {
57        Vec {
58            inflight: AtomicUsize::new(0),
59            buckets: [0; BUCKETS].map(|_| Bucket {
60                entries: AtomicPtr::new(ptr::null_mut()),
61            }),
62            count: AtomicUsize::new(0),
63        }
64    }
65
66    /// Constructs a new, empty `Vec<T>` with the specified capacity.
67    #[inline]
68    pub fn with_capacity(capacity: usize) -> Vec<T> {
69        let init = match capacity {
70            0 => 0,
71            // Initialize enough buckets for the n'th element to be inserted.
72            n => Location::of(n - 1).bucket,
73        };
74
75        let mut vec = Vec::new();
76        for (i, bucket) in vec.buckets[..=init].iter_mut().enumerate() {
77            // Initialize each bucket.
78            let len = Location::bucket_capacity(i);
79            // Safety: `Location::bucket_capacity` is non-zero.
80            *bucket = Bucket::from_ptr(unsafe { Bucket::alloc(len) });
81        }
82
83        vec
84    }

86    /// Returns the number of elements in the vector.
87    #[inline]
88    pub fn count(&self) -> usize {
89        // The `Acquire` here synchronizes with the `Release` increment
90        // when an entry is added to the vector.
91        self.count.load(Ordering::Acquire)
92    }

94    /// Returns a reference to the element at the given index.
95    #[inline]
96    pub fn get(&self, index: usize) -> Option<&T> {
97        self.get_raw(index)
98            // SAFETY: The entry returned by `get_raw` is guaranteed to be initialized.
99            .map(|entry| unsafe { (*entry).value_unchecked() })
100    }

102    /// Returns a mutable reference to the element at the given index.
103    #[inline]
104    pub fn get_mut(&mut self, index: usize) -> Option<&mut T> {
105        self.get_raw(index)
106            // SAFETY: The entry returned by `get_raw` is guaranteed to be
107            // initialized, and we have `&mut self`.
108            .map(|entry| unsafe { (*entry).value_unchecked_mut() })
109    }

111    /// Returns a pointer to the element at the given index.
112    ///
113    /// If `Some`, the returned pointer is guaranteed to be initialized.
114    #[inline]
115    fn get_raw(&self, index: usize) -> Option<*mut Entry<T>> {
116        let location = Location::of(index);
117
118        // Safety: `location.bucket` is always in bounds.
119        //
120        // The `Acquire` load here synchronizes with the `Release`
121        // store in `Vec::get_or_alloc`.
122        let entries = unsafe {
123            self.buckets
124                .get_unchecked(location.bucket)
125                .entries
126                .load(Ordering::Acquire)
127        };
128
129        // The bucket is uninitialized.
130        if entries.is_null() {
131            return None;
132        }
133
134        // Safety: `location.entry` is always in bounds for its bucket.
135        let entry = unsafe { entries.add(location.entry) };
136
137        // Safety: All entries are zero-initialized.
138        if unsafe { (*entry).active.load(Ordering::Acquire) } {
139            // Safety: The entry is active. Additionally, the `Acquire`
140            // load synchronizes with the `Release` store in `Vec::write`,
141            // ensuring the initialization happens-before this read.
142            return Some(entry);
143        }
144
145        // The entry is uninitialized.
146        None
147    }

149    /// Returns a reference to the element at the given index.
150    ///
151    /// # Safety
152    ///
153    /// The entry at `index` must be initialized.
154    #[inline]
155    pub unsafe fn get_unchecked(&self, index: usize) -> &T {
156        // Safety: Caller guarantees the bucket and entry are initialized.
157        unsafe {
158            let location = Location::of_unchecked(index);
159
160            // The `Acquire` load here synchronizes with the `Release`
161            // store in `Vec::get_or_alloc`.
162            let entry = self
163                .buckets
164                .get_unchecked(location.bucket)
165                .entries
166                .load(Ordering::Acquire)
167                .add(location.entry);
168
169            (*entry).value_unchecked()
170        }
171    }

173    /// Returns a mutable reference to the element at the given index.
174    ///
175    /// # Safety
176    ///
177    /// The entry at `index` must be initialized.
178    #[inline]
179    pub unsafe fn get_unchecked_mut(&mut self, index: usize) -> &mut T {
180        // Safety: Caller guarantees the bucket and entry are initialized.
181        unsafe {
182            let location = Location::of_unchecked(index);
183
184            let entry = self
185                .buckets
186                .get_unchecked_mut(location.bucket)
187                .entries
188                .read_mut()
189                .add(location.entry);
190
191            (*entry).value_unchecked_mut()
192        }
193    }

195    /// Returns a unique index for insertion.
196    #[inline]
197    fn next_index(&self) -> usize {
198        // The inflight counter cannot exceed `isize::MAX`, allowing it to catch capacity overflow.
199        //
200        // Note that the `Relaxed` ordering here is sufficient, as we only care about
201        // the index being unique and do not use it for synchronization.
202        let index = self.inflight.fetch_add(1, Ordering::Relaxed);
203        if index > MAX_INDEX {
204            // We could alternatively abort here, as `Arc` does. But we decrement and panic instead
205            // to keep in line with `Vec`'s behavior. Assuming that `isize::MAX` concurrent threads
206            // don't call this method, it is still impossible for it to overflow.
207            self.inflight.fetch_sub(1, Ordering::Relaxed);
208            panic!("capacity overflow");
209        }
210        index
211    }

213    /// Appends the element returned from the closure to the back of the vector
214    /// at the index represented by the `usize` passed to closure.
215    ///
216    /// This allows for use of the would-be index to be utilized within the
217    /// element.
218    #[inline]
219    pub fn push_with<F>(&self, f: F) -> usize
220    where
221        F: FnOnce(usize) -> T,
222    {
223        // Acquire a unique index to insert into.
224        let index = self.next_index();
225        let value = f(index);
226
227        // Safety: `next_index` is always in-bounds and unique.
228        unsafe { self.write(index, value) }
229    }

231    /// Appends an element to the back of the vector.
232    #[inline]
233    pub fn push(&self, value: T) -> usize {
234        // Safety: `next_index` is always in-bounds and unique.
235        unsafe { self.write(self.next_index(), value) }
236    }

238    /// Write an element at the given index.
239    ///
240    /// # Safety
241    ///
242    /// The index must be unique and in-bounds.
243    #[inline]
244    unsafe fn write(&self, index: usize, value: T) -> usize {
245        // Safety: Caller guarantees the entry is initialized.
246        let location = unsafe { Location::of_unchecked(index) };
247
248        // Eagerly allocate the next bucket if we are close to the end of this one.
249        if index == (location.bucket_len - (location.bucket_len >> 3)) {
250            if let Some(next_bucket) = self.buckets.get(location.bucket + 1) {
251                // Safety: Bucket lengths are non-zero.
252                unsafe { Vec::get_or_alloc(next_bucket, location.bucket_len << 1) };
253            }
254        }
255
256        // Safety: `location.bucket` is always in bounds.
257        let bucket = unsafe { self.buckets.get_unchecked(location.bucket) };
258
259        // The `Acquire` load here synchronizes with the `Release`
260        // store in `Vec::get_or_alloc`.
261        let mut entries = bucket.entries.load(Ordering::Acquire);
262
263        // The bucket has not been allocated yet.
264        if entries.is_null() {
265            // Safety: Bucket lengths are non-zero.
266            entries = unsafe { Vec::get_or_alloc(bucket, location.bucket_len) };
267        }
268
269        unsafe {
270            // Safety: We loaded the entries pointer with `Acquire` ordering,
271            // ensuring that it's initialization happens-before our access.
272            //
273            // Additionally, `location.entry` is always in bounds for its bucket.
274            let entry = &*entries.add(location.entry);
275
276            // Safety: We have unique access to this entry.
277            //
278            // 1. It is impossible for another thread to attempt a `push`
279            // to this location as we retrieved it from `next_index`.
280            //
281            // 2. Any thread trying to `get` this entry will see `!active`
282            // and will not try to access it.
283            entry
284                .slot
285                .with_mut(|slot| slot.write(MaybeUninit::new(value)));
286
287            // Let other threads know that this entry is active.
288            //
289            // Note that this `Release` write synchronizes with the `Acquire`
290            // load in `Vec::get`.
291            entry.active.store(true, Ordering::Release);
292        }
293
294        // Increase the element count.
295        //
296        // The `Release` here is not strictly necessary, but does
297        // allow users to use `count` for some sort of synchronization
298        // in terms of the number of initialized elements.
299        self.count.fetch_add(1, Ordering::Release);
300
301        // Return the index of the entry that we initialized.
302        index
303    }
304
305    /// Race to initialize a bucket.
306    ///
307    /// The returned pointer is guaranteed to be valid for access.
308    ///
309    /// Note that we avoid contention on bucket allocation by having a specified
310    /// writer eagerly allocate the next bucket.
311    ///
312    /// # Safety
313    ///
314    /// The provided length must be non-zero.
315    #[cold]
316    #[inline(never)]
317    unsafe fn get_or_alloc(bucket: &Bucket<T>, len: usize) -> *mut Entry<T> {
318        // Safety: Guaranteed by caller.
319        let entries = unsafe { Bucket::alloc(len) };
320
321        match bucket.entries.compare_exchange(
322            ptr::null_mut(),
323            entries,
324            // Establish synchronization with `Acquire` loads of the pointer.
325            Ordering::Release,
326            // If we lose the race, ensure that we synchronize with the initialization
327            // of the bucket that won.
328            Ordering::Acquire,
329        ) {
330            // We won the race.
331            Ok(_) => entries,
332
333            // We lost the race, deallocate our bucket and return the bucket that won.
334            Err(found) => unsafe {
335                Bucket::dealloc(entries, len);
336                found
337            },
338        }
339    }

341    /// Reserves capacity for at least `additional` more elements to be inserted in
342    /// the vector.
343    ///
344    /// The collection may reserve more space to avoid frequent reallocations.
345    pub fn reserve(&self, additional: usize) {
346        let len = self.count.load(Ordering::Acquire);
347        let mut location = Location::of(len.checked_add(additional).unwrap_or(MAX_INDEX));
348
349        // Allocate buckets starting from the bucket at `len + additional` and
350        // working our way backwards.
351        loop {
352            // Safety: `location.bucket` is always in bounds.
353            let bucket = unsafe { self.buckets.get_unchecked(location.bucket) };
354
355            // Reached an initialized bucket, we're done.
356            //
357            // `Relaxed` is sufficient here as we never access the bucket
358            // and only ensure that it is initialized.
359            if !bucket.entries.load(Ordering::Relaxed).is_null() {
360                break;
361            }
362
363            // Allocate the bucket.
364            //
365            // Safety: Bucket lengths are non-zero.
366            unsafe { Vec::get_or_alloc(bucket, location.bucket_len) };
367
368            // Reached the first bucket, we're done.
369            if location.bucket == 0 {
370                break;
371            }
372
373            location.bucket -= 1;
374            location.bucket_len = Location::bucket_capacity(location.bucket);
375        }
376    }

378    // Returns an iterator over the vector.
379    #[inline]
380    pub fn iter(&self) -> Iter {
381        Iter {
382            index: 0,
383            yielded: 0,
384            // Snapshot the number of potentially active entries when the iterator is
385            // created, so we know how many entries we have to check. The alternative
386            // is to always check every single entry and/or bucket. Note that yielding
387            // `self.count` entries is not enough; we may end up yielding the *wrong*
388            // `self.count` entries, as the gaps in-between the entries we want to yield
389            // may get filled concurrently.
390            inflight: core::cmp::min(self.inflight.load(Ordering::Relaxed), MAX_INDEX),
391        }
392    }
393
394    /// Clear every element in the vector.
395    #[inline]
396    pub fn clear(&mut self) {
397        let mut iter = self.iter();
398
399        // Consume and reset every entry in the vector.
400        while iter.next_owned(self).is_some() {}
401
402        // Reset the count.
403        self.count.store(0, Ordering::Relaxed);
404        self.inflight.store(0, Ordering::Relaxed);
405    }

418impl<T> Drop for Vec<T> {
419    fn drop(&mut self) {
420        for (i, bucket) in self.buckets.iter_mut().enumerate() {
421            let entries = bucket.entries.read_mut();
422
423            if entries.is_null() {
424                break;
425            }
426
427            let len = Location::bucket_capacity(i);
428
429            // Safety: We have `&mut self` and verified that this bucket is
430            // initialized.
431            unsafe { Bucket::dealloc(entries, len) }
432        }
433    }
434}

449/// A possibly uninitialized entry in the vector.
450struct Entry<T> {
451    /// A flag indicating whether or not this entry is initialized.
452    active: AtomicBool,
453
454    /// The entry's value.
455    slot: UnsafeCell<MaybeUninit<T>>,
456}
457
458impl<T> Bucket<T> {
459    /// Create a `Bucket` from the given `Entry` pointer.
460    fn from_ptr(entries: *mut Entry<T>) -> Bucket<T> {
461        Bucket {
462            entries: AtomicPtr::new(entries),
463        }
464    }
465
466    /// Allocate an array of entries of the specified length.
467    ///
468    /// # Safety
469    ///
470    /// The provided length must be non-zero.
471    #[cfg(not(loom))]
472    unsafe fn alloc(len: usize) -> *mut Entry<T> {
473        let layout = Layout::array::<Entry<T>>(len).unwrap();
474
475        // Note that this sets the `active` flag to `false`.
476        //
477        // Safety: Caller guarantees that `layout` has a non-zero size, and
478        // `AtomicBool`, `UnsafeCell`, and `MaybeUninit` are zeroable types.
479        let ptr = unsafe { alloc::alloc::alloc_zeroed(layout) };
480
481        // Handle allocation errors.
482        if ptr.is_null() {
483            alloc::alloc::handle_alloc_error(layout);
484        }
485
486        ptr.cast::<Entry<T>>()
487    }
488
489    /// Allocate an array of entries of the specified length.
490    #[cfg(loom)]
491    unsafe fn alloc(len: usize) -> *mut Entry<T> {
492        // Note we cannot use `alloc_zeroed` for Loom types.
493        let entries = (0..len)
494            .map(|_| Entry::<T> {
495                slot: UnsafeCell::new(MaybeUninit::uninit()),
496                active: AtomicBool::new(false),
497            })
498            .collect::<Box<[Entry<_>]>>();
499
500        Box::into_raw(entries) as *mut Entry<T>
501    }
502
503    /// Deallocate a bucket of the specified capacity.
504    ///
505    /// # Safety
506    ///
507    /// The safety requirements of `slice::from_raw_parts_mut` and
508    /// `Box::from_raw`. The pointer must be a valid, owned pointer
509    /// to an array of entries of the provided length.
510    unsafe fn dealloc(entries: *mut Entry<T>, len: usize) {
511        // Safety: Guaranteed by caller.
512        unsafe { drop(Box::from_raw(slice::from_raw_parts_mut(entries, len))) }
513    }
514}

516impl<T> Entry<T> {
517    /// Returns a reference to the value in this entry.
518    ///
519    /// # Safety
520    ///
521    /// The value must be initialized.

524        // Safety: Guaranteed by caller.
525        self.slot.with(|slot| unsafe { (*slot).assume_init_ref() })

527
528    /// Returns a mutable reference to the value in this entry.

530    // # Safety
531    //
532    // The value must be initialized.
533    #[inline]
534    unsafe fn value_unchecked_mut(&mut self) -> &mut T {

536        self.slot
537            .with_mut(|slot| unsafe { (*slot).assume_init_mut() })

541impl<T> Drop for Entry<T> {
542    fn drop(&mut self) {
543        if self.active.read_mut() {
544            // Safety: We have `&mut self` and verifid that the value is initialized.
545            unsafe { ptr::drop_in_place(self.slot.with_mut(|slot| (*slot).as_mut_ptr())) }
546        }

550/// The location of an entry in the bucket array.
551#[derive(Debug, Clone)]
552struct Location {
553    /// The index of the bucket.
554    bucket: usize,
555
556    /// The length of the bucket at the above index.
557    bucket_len: usize,
558
559    /// The index of the entry in the bucket.
560    entry: usize,
561}
562
563/// The number of entries that are skipped from the start of a vector.
564///
565/// Index calculations assume that buckets are of sizes `[2^0, 2^1, ..., 2^62]`.
566/// To skip shorter buckets and avoid unnecessary location, the zeroeth entry
567/// index is remapped to a larger index (`2^0 + ... + 2^4 = 31`).
568const ZERO_ENTRY: usize = 31;
569
570/// The number of buckets that are skipped from the start of a vector.
571///
572/// This is the index that the zeroeth bucket index is remapped to (`5`).
573const ZERO_BUCKET: usize = (usize::BITS - ZERO_ENTRY.leading_zeros()) as usize;
574
575/// The number of buckets in a vector.
576const BUCKETS: usize = (usize::BITS as usize) - 1 - ZERO_BUCKET;
577
578/// The maximum index of an element in the vector.
579///
580/// Note that capacity of the vector is:
581/// `2^ZERO_BUCKET + ... + 2^62 = isize::MAX - ZERO_INDEX`.
582///
583/// We limit at `isize::MAX` instead of `usize::MAX` to allow checking for overflows in
584/// `next_index` using only `fetch_add`. In practice, you won't be able to make a `Vec` with this
585/// many elements unless it's a `Vec<()>` (since each entry adds one byte to `T`), and we don't
586/// particularly care to support that case.
587const MAX_INDEX: usize = (isize::MAX as usize) - ZERO_ENTRY - 1;
588
589impl Location {
590    /// Returns the location of a given entry in a vector.
591    ///
592    /// This function will panic if the entry is greater than `MAX_INDEX`.
593    #[inline]
594    fn of(index: usize) -> Location {
595        if index > MAX_INDEX {
596            panic!("index out of bounds");
597        }
598
599        Location::of_raw(index + ZERO_ENTRY)
600    }
601
602    /// Returns the location of a given entry in a vector, without bounds checks.
603    ///
604    /// # Safety
605    ///
606    /// The index must be in-bounds.
607    #[inline]
608    unsafe fn of_unchecked(index: usize) -> Location {
609        // Note: This can lead to unsoundness if it wraps.
610        Location::of_raw(index + ZERO_ENTRY)
611    }
612
613    /// Returns the location of the entry at index `index - ZERO_INDEX` in a vector.
614    #[inline]
615    fn of_raw(index: usize) -> Location {
616        // Calculate the bucket index based on ⌊log2(index)⌋.
617        let bucket = BUCKETS - ((index + 1).leading_zeros() as usize);
618        let bucket_len = Location::bucket_capacity(bucket);
619
620        // Offset the absolute index by the capacity of the preceding buckets.
621        let entry = index - (bucket_len - 1);
622
623        Location {
624            bucket,
625            bucket_len,
626            entry,
627        }
628    }
629
630    /// Returns the capacity of the bucket at the given index.
631    #[inline]
632    fn bucket_capacity(bucket: usize) -> usize {
633        1 << (bucket + ZERO_BUCKET)
634    }
635}

637/// An iterator over the elements of a [`Vec<T>`].
638#[derive(Clone)]
639pub struct Iter {
640    yielded: usize,
641    inflight: usize,
642    index: usize,
643}
644
645impl Iter {
646    /// Returns a pointer to the next entry in the iterator.
647    ///
648    /// Note that the iterator is fused, and any returned entries are guaranteed
649    /// to be initialized.
650    #[inline]
651    fn next<T>(&mut self, vec: &Vec<T>) -> Option<(usize, *mut Entry<T>)> {
652        // Checked every entry in the vector that was potentially initialized when
653        // this iterator was created.
654        while self.index < self.inflight {
655            let index = self.index;
656            self.index += 1;
657
658            let Some(entry) = vec.get_raw(index) else {
659                continue;
660            };
661
662            return Some((index, entry));
663        }
664
665        None
666    }
667
668    /// Returns a shared reference to the next entry in the iterator.
669    #[inline]
670    pub fn next_shared<'v, T>(&mut self, vec: &'v Vec<T>) -> Option<(usize, &'v T)> {
671        self.next(vec)
672            // Safety: `Iter::next` guarantees that the entry is initialized.
673            .map(|(index, entry)| (index, unsafe { (*entry).value_unchecked() }))
674    }
675
676    /// Returns an owned reference to the next entry in the iterator, resetting the entry in
677    /// the vector.
678    #[inline]
679    pub fn next_owned<T>(&mut self, vec: &mut Vec<T>) -> Option<T> {
680        self.next(vec).map(|(_, entry)| {
681            // Safety: `Iter::next` guarantees that the entry is initialized, and we have `&mut Vec<T>`.
682            let entry = unsafe { &mut *entry };
683
684            // Mark the entry as uninitialized so it is not accessed after this.
685            entry.active.write_mut(false);
686
687            // Safety: `Iter::next` only yields initialized entries.
688            unsafe {
689                let slot = entry.slot.with_mut(|slot| &mut *slot);
690                mem::replace(slot, MaybeUninit::uninit()).assume_init()
691            }
692        })
693    }
694
695    /// Returns the number of elements that have been yielded by this iterator.
696    #[inline]
697    pub fn yielded(&self) -> usize {
698        self.yielded
699    }
700}
