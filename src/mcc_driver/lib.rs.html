1//! Michael's C Compiler — driver library
2//!
3//! This crate provides the command-line interface and orchestration for the
4//! `mcc` compiler. It wires together the core stages from the `mcc` crate,
5//! manages temporary files, captures diagnostics between stages, and exposes a
6//! small callback API for tooling and testing.
7//!
8//! What this crate offers
9//! ----------------------
10//! - A `main()` entrypoint for the `mcc` binary: [`cli::main`]
11//! - A stage runner with callbacks: [`run`] and the [`Callbacks`] trait
12//! - A configuration object for compilation sessions: [`Config`]
13//! - An outcome type capturing success, error, or early return: [`Outcome`]
14//!
15//! Running the pipeline with callbacks
16//! -----------------------------------
17//! The [`run`] function executes the full pipeline:
18//! preprocessing → parsing → lowering → codegen → rendering → assemble+link.
19//!
20//! You can implement [`Callbacks`] to observe intermediate artifacts, collect
21//! diagnostics, or stop the pipeline early:
22//!
23//! ```rust
24//! use std::ops::ControlFlow;
25//! use mcc_driver::{Callbacks, Config, Outcome};
26//! use mcc::{Ast, Text, diagnostics::Diagnostics, codegen::asm, lowering::tacky};
27//!
28//! struct Noop;
29//! impl Callbacks for Noop {
30//!     type Output = ();
31//!
32//!     fn after_parse<'db>(
33//!         &mut self,
34//!         _db: &'db dyn mcc::Db,
35//!         _source_file: mcc::SourceFile,
36//!         _ast: Ast<'db>,
37//!         _diags: Vec<&Diagnostics>,
38//!     ) -> ControlFlow<Self::Output> {
39//!         ControlFlow::Continue(())
40//!     }
41//!
42//!     fn after_lower<'db>(
43//!         &mut self,
44//!         _db: &'db dyn mcc::Db,
45//!         _tacky: tacky::Program<'db>,
46//!         _diags: Vec<&Diagnostics>,
47//!     ) -> ControlFlow<Self::Output> { ControlFlow::Continue(()) }
48//!
49//!     fn after_codegen<'db>(
50//!         &mut self,
51//!         _db: &'db dyn mcc::Db,
52//!         _asm: asm::Program<'db>,
53//!         _diags: Vec<&Diagnostics>,
54//!     ) -> ControlFlow<Self::Output> { ControlFlow::Continue(()) }
55//!
56//!     fn after_render_assembly(
57//!         &mut self,
58//!         _db: &dyn mcc::Db,
59//!         _asm: Text,
60//!         _diags: Vec<&Diagnostics>,
61//!     ) -> ControlFlow<Self::Output> { ControlFlow::Continue(()) }
62//! }
63//!
64//! // In your application:
65//! // let outcome = mcc_driver::run(&mut Noop, config);
66//! // outcome.to_result()?;
67//! ```
68//!
69//! CLI entrypoint
70//! --------------
71//! The binary uses [`cli::main`] to parse flags (target triple, output path,
72//! color, stop-at-stage switches, etc.), set up tracing, and delegate to
73//! [`run`]. See `crates/mcc-driver/src/cli.rs` for details.
74//!
75
76mod callbacks;
77mod cli;
78
79pub use crate::{
80    callbacks::{Callbacks, Config, Outcome, run},
81    cli::main,
82};
