8/// For an entity struct `Foo` with fields `f1: T1, ..., fN: TN`, we generate...
9///
10/// * the "id struct" `struct Foo(salsa::Id)`
11/// * the entity ingredient, which maps the id fields to the `Id`
12/// * for each value field, a function ingredient
13pub(crate) fn input(
14    args: proc_macro::TokenStream,
15    input: proc_macro::TokenStream,
16) -> proc_macro::TokenStream {
17    let args = syn::parse_macro_input!(args as InputArgs);
18    let hygiene = Hygiene::from1(&input);
19    let struct_item = parse_macro_input!(input as syn::ItemStruct);
20    let m = Macro {
21        hygiene,
22        args,
23        struct_item,
24    };
25    match m.try_macro() {
26        Ok(v) => v.into(),
27        Err(e) => token_stream_with_error(input, e),
28    }
29}
