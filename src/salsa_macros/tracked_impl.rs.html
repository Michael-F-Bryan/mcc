12pub(crate) fn tracked_impl(
13    args: proc_macro::TokenStream,
14    item: syn::ItemImpl,
15) -> syn::Result<TokenStream> {
16    let hygiene = Hygiene::from2(&item);
17    let _: Nothing = syn::parse(args)?;
18    let m = Macro { hygiene };
19    let generated = m.try_generate(item)?;
20    Ok(generated)
21}
22
23struct Macro {
24    hygiene: Hygiene,
25}
26
27struct AssociatedFunctionArguments<'syn> {
28    self_token: Option<&'syn syn::token::SelfValue>,
29    db_ty: &'syn syn::Type,
30    db_ident: &'syn syn::Ident,
31    db_lt: Option<&'syn syn::Lifetime>,
32    input_ids: Vec<syn::Ident>,
33    input_tys: Vec<&'syn syn::Type>,
34    output_ty: syn::Type,
35}

37impl Macro {
38    fn try_generate(&self, mut impl_item: syn::ItemImpl) -> syn::Result<TokenStream> {
39        let mut member_items = std::mem::take(&mut impl_item.items);
40        let member_idents: HashSet<_> = member_items
41            .iter()
42            .filter_map(|item| match item {
43                syn::ImplItem::Const(it) => Some(it.ident.clone()),
44                syn::ImplItem::Fn(it) => Some(it.sig.ident.clone()),
45                syn::ImplItem::Type(it) => Some(it.ident.clone()),
46                syn::ImplItem::Macro(_) => None,
47                syn::ImplItem::Verbatim(_) => None,
48                _ => None,
49            })
50            .collect();
51        for member_item in &mut member_items {
52            self.modify_member(&impl_item, member_item, &member_idents)?;
53        }
54        impl_item.items = member_items;
55        Ok(crate::debug::dump_tokens(
56            format!("impl {:?}", impl_item.self_ty),
57            impl_item.into_token_stream(),
58        ))
59    }

61    #[allow(non_snake_case)]
62    fn modify_member(
63        &self,
64        impl_item: &syn::ItemImpl,
65        member_item: &mut syn::ImplItem,
66        member_idents: &HashSet<syn::Ident>,
67    ) -> syn::Result<()> {
68        let syn::ImplItem::Fn(fn_item) = member_item else {
69            return Ok(());
70        };
71
72        let self_ty = &*impl_item.self_ty;
73
74        let Some(tracked_attr_index) = fn_item.attrs.iter().position(|a| self.is_tracked_attr(a))
75        else {
76            return Ok(());
77        };
78
79        let trait_ = match &impl_item.trait_ {
80            Some((None, path, _)) => Some((path, member_idents)),
81            _ => None,
82        };
83        let mut change = ChangeSelfPath::new(self_ty, trait_);
84        change.visit_impl_item_fn_mut(fn_item);
85
86        let mut salsa_tracked_attr = fn_item.attrs.remove(tracked_attr_index);
87        let mut args: FnArgs = match &salsa_tracked_attr.meta {
88            syn::Meta::Path(..) => Default::default(),
89            _ => salsa_tracked_attr.parse_args()?,
90        };
91        if args.self_ty.is_none() {
92            // If the user did not specify a self_ty, we use the impl's self_ty
93            args.self_ty = Some(self_ty.clone());
94        }
95        salsa_tracked_attr.meta = syn::Meta::List(syn::MetaList {
96            path: salsa_tracked_attr.path().clone(),
97            delimiter: syn::MacroDelimiter::Paren(syn::token::Paren::default()),
98            tokens: quote! {#args},
99        });
100
101        let InnerTrait = self.hygiene.ident("InnerTrait");
102        let inner_fn_name = self.hygiene.ident(&fn_item.sig.ident.to_string());
103
104        let AssociatedFunctionArguments {
105            self_token,
106            db_ty,
107            db_ident,
108            db_lt,
109            input_ids,
110            input_tys,
111            output_ty,
112        } = self.validity_check(impl_item, fn_item)?;
113
114        let mut inner_fn = fn_item.clone();
115        inner_fn.vis = syn::Visibility::Inherited;
116        inner_fn.sig.ident = inner_fn_name.clone();
117
118        // Construct the body of the method or associated function
119
120        let block = if let Some(self_token) = self_token {
121            parse_quote!({
122                salsa::plumbing::setup_tracked_method_body! {
123                    salsa_tracked_attr: #salsa_tracked_attr,
124                    self: #self_token,
125                    self_ty: #self_ty,
126                    db_lt: #db_lt,
127                    db: #db_ident,
128                    db_ty: (#db_ty),
129                    input_ids: [#(#input_ids),*],
130                    input_tys: [#(#input_tys),*],
131                    output_ty: #output_ty,
132                    inner_fn_name: #inner_fn_name,
133                    inner_fn: #inner_fn,
134
135                    // Annoyingly macro-rules hygiene does not extend to items defined in the macro.
136                    // We have the procedural macro generate names for those items that are
137                    // not used elsewhere in the user's code.
138                    unused_names: [
139                        #InnerTrait,
140                    ]
141                }
142            })
143        } else {
144            parse_quote!({
145                salsa::plumbing::setup_tracked_assoc_fn_body! {
146                    salsa_tracked_attr: #salsa_tracked_attr,
147                    self_ty: #self_ty,
148                    db_lt: #db_lt,
149                    db: #db_ident,
150                    db_ty: (#db_ty),
151                    input_ids: [#(#input_ids),*],
152                    input_tys: [#(#input_tys),*],
153                    output_ty: #output_ty,
154                    inner_fn_name: #inner_fn_name,
155                    inner_fn: #inner_fn,
156
157                    // Annoyingly macro-rules hygiene does not extend to items defined in the macro.
158                    // We have the procedural macro generate names for those items that are
159                    // not used elsewhere in the user's code.
160                    unused_names: [
161                        #InnerTrait,
162                    ]
163                }
164            })
165        };
166
167        // Update the method that will actually appear in the impl to have the new body
168        // and its true return type
169        let db_lt = db_lt.cloned();
170        self.update_return_type(&mut fn_item.sig, &args, &db_lt)?;
171        fn_item.block = block;
172
173        Ok(())
174    }

176    fn validity_check<'syn>(
177        &self,
178        impl_item: &'syn syn::ItemImpl,
179        fn_item: &'syn syn::ImplItemFn,
180    ) -> syn::Result<AssociatedFunctionArguments<'syn>> {
181        let db_lt = self.extract_db_lifetime(impl_item, fn_item)?;
182
183        let is_method = matches!(&fn_item.sig.inputs[0], syn::FnArg::Receiver(_));
184
185        let (self_token, db_input_index, skipped_inputs) = if is_method {
186            (Some(self.check_self_argument(fn_item)?), 1, 2)
187        } else {
188            (None, 0, 1)
189        };
190
191        let (db_ident, db_ty) = self.check_db_argument(&fn_item.sig.inputs[db_input_index])?;
192
193        let input_ids: Vec<syn::Ident> =
194            crate::fn_util::input_ids(&self.hygiene, &fn_item.sig, skipped_inputs);
195        let input_tys = crate::fn_util::input_tys(&fn_item.sig, skipped_inputs)?;
196        let output_ty = crate::fn_util::output_ty(db_lt, &fn_item.sig)?;
197
198        Ok(AssociatedFunctionArguments {
199            self_token,
200            db_ident,
201            db_lt,
202            db_ty,
203            input_ids,
204            input_tys,
205            output_ty,
206        })
207    }

209    fn is_tracked_attr(&self, attr: &syn::Attribute) -> bool {
210        if attr.path().segments.len() != 2 {
211            return false;
212        }
213
214        let seg0 = &attr.path().segments[0];
215        let seg1 = &attr.path().segments[1];
216
217        seg0.ident == "salsa"
218            && seg1.ident == "tracked"
219            && seg0.arguments.is_empty()
220            && seg1.arguments.is_empty()
221    }

223    fn extract_db_lifetime<'syn>(
224        &self,
225        impl_item: &'syn syn::ItemImpl,
226        fn_item: &'syn syn::ImplItemFn,
227    ) -> syn::Result<Option<&'syn syn::Lifetime>> {
228        // Either the impl XOR the fn can have generics, and it must be at most a lifetime
229        let mut db_lt = None;
230        for param in impl_item
231            .generics
232            .params
233            .iter()
234            .chain(fn_item.sig.generics.params.iter())
235        {
236            match param {
237                syn::GenericParam::Lifetime(lt) => {
238                    if db_lt.is_none() {
239                        if let Some(bound) = lt.bounds.iter().next() {
240                            return Err(syn::Error::new_spanned(
241                                bound,
242                                "lifetime parameters on tracked methods must not have bounds",
243                            ));
244                        }
245
246                        db_lt = Some(&lt.lifetime);
247                    } else {
248                        return Err(syn::Error::new_spanned(
249                            param,
250                            "tracked method already has a lifetime parameter in scope",
251                        ));
252                    }
253                }
254                _ => {
255                    return Err(syn::Error::new_spanned(
256                        param,
257                        "tracked methods cannot have non-lifetime generic parameters",
258                    ));
259                }
260            }
261        }
262
263        Ok(db_lt)
264    }
265
266    fn check_self_argument<'syn>(
267        &self,
268        fn_item: &'syn syn::ImplItemFn,
269    ) -> syn::Result<&'syn syn::token::SelfValue> {
270        if fn_item.sig.inputs.is_empty() {
271            return Err(syn::Error::new_spanned(
272                &fn_item.sig.ident,
273                "tracked methods must have arguments",
274            ));
275        }
276
277        let syn::FnArg::Receiver(syn::Receiver {
278            attrs: _,
279            self_token,
280            reference,
281            mutability: _,
282            colon_token,
283            ty: _,
284        }) = &fn_item.sig.inputs[0]
285        else {
286            return Err(syn::Error::new_spanned(
287                &fn_item.sig.inputs[0],
288                "tracked methods must take a `self` argument",
289            ));
290        };
291
292        if let Some(colon_token) = colon_token {
293            return Err(syn::Error::new_spanned(
294                colon_token,
295                "tracked method's `self` argument must not have an explicit type",
296            ));
297        }
298
299        if let Some((and_token, _)) = reference {
300            return Err(syn::Error::new_spanned(
301                and_token,
302                "tracked methods's first argument must be declared as `self`, not `&self` or `&mut self`",
303            ));
304        }
305
306        Ok(self_token)
307    }

309    fn check_db_argument<'syn>(
310        &self,
311        input: &'syn syn::FnArg,
312    ) -> syn::Result<(&'syn syn::Ident, &'syn syn::Type)> {
313        let syn::FnArg::Typed(typed) = input else {
314            return Err(syn::Error::new_spanned(
315                input,
316                "tracked methods must take a database parameter",
317            ));
318        };
319
320        let syn::Pat::Ident(db_pat_ident) = &*typed.pat else {
321            return Err(syn::Error::new_spanned(
322                &typed.pat,
323                "database parameter must have a simple name",
324            ));
325        };
326
327        let db_ident = &db_pat_ident.ident;
328        let db_ty = &*typed.ty;
329
330        Ok((db_ident, db_ty))
331    }
332
333    fn update_return_type(
334        &self,
335        sig: &mut syn::Signature,
336        args: &FnArgs,
337        db_lt: &Option<syn::Lifetime>,
338    ) -> syn::Result<()> {
339        if let Some(returns) = &args.returns {
340            if let syn::ReturnType::Type(_, t) = &mut sig.output {
341                if returns == "copy" || returns == "clone" {
342                    // leave as is
343                } else if returns == "ref" {
344                    **t = parse_quote!(& #db_lt #t)
345                } else if returns == "deref" {
346                    **t = parse_quote!(& #db_lt <#t as ::core::ops::Deref>::Target)
347                } else if returns == "as_ref" {
348                    **t = parse_quote!(<#t as ::salsa::SalsaAsRef>::AsRef<#db_lt>)
349                } else if returns == "as_deref" {
350                    **t = parse_quote!(<#t as ::salsa::SalsaAsDeref>::AsDeref<#db_lt>)
351                } else {
352                    return Err(syn::Error::new_spanned(
353                        returns,
354                        format!("Unknown returns mode `{returns}`"),
355                    ));
356                }
357            } else {
358                return Err(syn::Error::new_spanned(
359                    returns,
360                    "returns attribute requires explicit return type",
361                ));
362            };
363        }
364        Ok(())
365    }
