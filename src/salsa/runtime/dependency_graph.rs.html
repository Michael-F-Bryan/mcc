12#[derive(Debug, Default)]
13pub(super) struct DependencyGraph {
14    /// A `(K -> V)` pair in this map indicates that the runtime
15    /// `K` is blocked on some query executing in the runtime `V`.
16    /// This encodes a graph that must be acyclic (or else deadlock
17    /// will result).
18    edges: FxHashMap<ThreadId, edge::Edge>,
19
20    /// Encodes the `ThreadId` that are blocked waiting for the result
21    /// of a given query.
22    query_dependents: FxHashMap<DatabaseKeyIndex, SmallVec<[ThreadId; 4]>>,
23
24    /// When a key K completes which had dependent queries Qs blocked on it,
25    /// it stores its `WaitResult` here. As they wake up, each query Q in Qs will
26    /// come here to fetch their results.
27    wait_results: FxHashMap<ThreadId, WaitResult>,
28}
29
30impl DependencyGraph {
31    /// True if `from_id` depends on `to_id`.
32    ///
33    /// (i.e., there is a path from `from_id` to `to_id` in the graph.)
34    pub(super) fn depends_on(&self, from_id: ThreadId, to_id: ThreadId) -> bool {
35        let mut p = from_id;
36        while let Some(q) = self.edges.get(&p).map(|edge| edge.blocked_on_id) {
37            if q == to_id {
38                return true;
39            }
40
41            p = q;
42        }
43        p == to_id
44    }
45
46    /// Modifies the graph so that `from_id` is blocked
47    /// on `database_key`, which is being computed by
48    /// `to_id`.
49    ///
50    /// For this to be reasonable, the lock on the
51    /// results table for `database_key` must be held.
52    /// This ensures that computing `database_key` doesn't
53    /// complete before `block_on` executes.
54    ///
55    /// Preconditions:
56    /// * No path from `to_id` to `from_id`
57    ///   (i.e., `me.depends_on(to_id, from_id)` is false)
58    /// * `held_mutex` is a read lock (or stronger) on `database_key`
59    pub(super) fn block_on<QueryMutexGuard>(
60        mut me: MutexGuard<'_, Self>,
61        from_id: ThreadId,
62        database_key: DatabaseKeyIndex,
63        to_id: ThreadId,
64        query_mutex_guard: QueryMutexGuard,
65    ) -> WaitResult {
66        let cvar = std::pin::pin!(EdgeCondvar::default());
67        let cvar = cvar.as_ref();
68        // SAFETY: We are blocking until the result is removed from `DependencyGraph::wait_results`
69        // at which point the `edge` won't signal the condvar anymore.
70        // As such we are keeping the cond var alive until the reference in the edge drops.
71        unsafe { me.add_edge(from_id, database_key, to_id, cvar) };
72
73        // Release the mutex that prevents `database_key`
74        // from completing, now that the edge has been added.
75        drop(query_mutex_guard);
76
77        loop {
78            if let Some(result) = me.wait_results.remove(&from_id) {
79                debug_assert!(!me.edges.contains_key(&from_id));
80                return result;
81            }
82            me = cvar.wait(me);
83        }
84    }
85
86    /// Helper for `block_on`: performs actual graph modification
87    /// to add a dependency edge from `from_id` to `to_id`, which is
88    /// computing `database_key`.
89    ///
90    /// # Safety
91    ///
92    /// The caller needs to keep the referent of `cvar` alive until the corresponding
93    /// [`Self::wait_results`] entry has been inserted.
94    unsafe fn add_edge(
95        &mut self,
96        from_id: ThreadId,
97        database_key: DatabaseKeyIndex,
98        to_id: ThreadId,
99        cvar: Pin<&EdgeCondvar>,
100    ) {
101        assert_ne!(from_id, to_id);
102        debug_assert!(!self.edges.contains_key(&from_id));
103        debug_assert!(!self.depends_on(to_id, from_id));
104        // SAFETY: The caller is responsible for ensuring that the `EdgeGuard` outlives the `Edge`.
105        let edge = unsafe { edge::Edge::new(to_id, cvar) };
106        self.edges.insert(from_id, edge);
107        self.query_dependents
108            .entry(database_key)
109            .or_default()
110            .push(from_id);
111    }
112
113    /// Invoked when runtime `to_id` completes executing
114    /// `database_key`.
115    pub(super) fn unblock_runtimes_blocked_on(
116        &mut self,
117        database_key: DatabaseKeyIndex,
118        wait_result: WaitResult,
119    ) {
120        let dependents = self
121            .query_dependents
122            .remove(&database_key)
123            .unwrap_or_default();
124
125        for from_id in dependents {
126            self.unblock_runtime(from_id, wait_result);
127        }
128    }
129
130    /// Unblock the runtime with the given id with the given wait-result.
131    /// This will cause it resume execution (though it will have to grab
132    /// the lock on this data structure first, to recover the wait result).
133    fn unblock_runtime(&mut self, id: ThreadId, wait_result: WaitResult) {
134        let edge = self.edges.remove(&id).expect("not blocked");
135        self.wait_results.insert(id, wait_result);
136
137        // Now that we have inserted the `wait_results`,
138        // notify the thread.
139        edge.notify();
140    }
141}

143mod edge {
144    use crate::sync::thread::ThreadId;
145    use crate::sync::{Condvar, MutexGuard};
146
147    use std::pin::Pin;
148
149    #[derive(Default, Debug)]
150    pub(super) struct EdgeCondvar {
151        condvar: Condvar,
152        _phantom_pin: std::marker::PhantomPinned,
153    }
154
155    impl EdgeCondvar {
156        #[inline]
157        pub(super) fn wait<'a, T>(&self, mutex_guard: MutexGuard<'a, T>) -> MutexGuard<'a, T> {
158            self.condvar.wait(mutex_guard)
159        }
160    }
161
162    #[derive(Debug)]
163    pub(super) struct Edge {
164        pub(super) blocked_on_id: ThreadId,
165
166        /// Signalled whenever a query with dependents completes.
167        /// Allows those dependents to check if they are ready to unblock.
168        // condvar: unsafe<'stack_frame> Pin<&'stack_frame Condvar>,
169        condvar: Pin<&'static EdgeCondvar>,
170    }
171
172    impl Edge {
173        /// # SAFETY
174        ///
175        /// The caller must ensure that the [`EdgeCondvar`] is kept alive until the [`Edge`] is dropped.
176        pub(super) unsafe fn new(blocked_on_id: ThreadId, condvar: Pin<&EdgeCondvar>) -> Self {
177            Self {
178                blocked_on_id,
179                // SAFETY: The caller is responsible for ensuring that the `EdgeCondvar` outlives the `Edge`.
180                condvar: unsafe {
181                    std::mem::transmute::<Pin<&EdgeCondvar>, Pin<&'static EdgeCondvar>>(condvar)
182                },
183            }
184        }
185
186        #[inline]
187        pub(super) fn notify(self) {
188            self.condvar.condvar.notify_one();
189        }
190    }
191}
