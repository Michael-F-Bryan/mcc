31/// Trait that defines the key properties of a tracked struct.
32///
33/// Implemented by the `#[salsa::tracked]` macro when applied
34/// to a struct.
35pub trait Configuration: Sized + 'static {
36    const LOCATION: crate::ingredient::Location;
37
38    /// The debug name of the tracked struct.
39    const DEBUG_NAME: &'static str;
40
41    /// The debug names of any tracked fields.
42    const TRACKED_FIELD_NAMES: &'static [&'static str];
43
44    /// The relative indices of any tracked fields.
45    const TRACKED_FIELD_INDICES: &'static [usize];
46
47    /// A (possibly empty) tuple of the fields for this struct.
48    type Fields<'db>: Send + Sync;
49
50    /// A array of [`Revision`][] values, one per each of the tracked value fields.
51    /// When a struct is re-recreated in a new revision, the corresponding
52    /// entries for each field are updated to the new revision if their
53    /// values have changed (or if the field is marked as `#[no_eq]`).
54    type Revisions: Send + Sync + Index<usize, Output = Revision>;
55
56    type Struct<'db>: Copy + FromId + AsId;
57
58    fn untracked_fields(fields: &Self::Fields<'_>) -> impl Hash;
59
60    /// Create a new value revision array where each element is set to `current_revision`.
61    fn new_revisions(current_revision: Revision) -> Self::Revisions;
62
63    /// Update the field data and, if the value has changed,
64    /// the appropriate entry in the `revisions` array (tracked fields only).
65    ///
66    /// Returns `true` if any untracked field was updated and
67    /// the struct should be considered re-created.
68    ///
69    /// # Safety
70    ///
71    /// Requires the same conditions as the `maybe_update`
72    /// method on [the `Update` trait](`crate::update::Update`).
73    ///
74    /// In short, requires that `old_fields` be a pointer into
75    /// storage from a previous revision.
76    /// It must meet its validity invariant.
77    /// Owned content must meet safety invariant.
78    /// `*mut` here is not strictly needed;
79    /// it is used to signal that the content
80    /// is not guaranteed to recursively meet
81    /// its safety invariant and
82    /// hence this must be dereferenced with caution.
83    ///
84    /// Ensures that `old_fields` is fully updated and valid
85    /// after it returns and that `revisions` has been updated
86    /// for any field that changed.
87    unsafe fn update_fields<'db>(
88        current_revision: Revision,
89        revisions: &mut Self::Revisions,
90        old_fields: *mut Self::Fields<'db>,
91        new_fields: Self::Fields<'db>,
92    ) -> bool;
93}

145/// Created for each tracked struct.
146///
147/// This ingredient only stores the "id" fields. It is a kind of "dressed up" interner;
148/// the active query + values of id fields are hashed to create the tracked
149/// struct id. The value fields are stored in [`crate::function::IngredientImpl`]
150/// instances keyed by the tracked struct id.
151///
152/// Unlike normal interned values, tracked struct indices can be deleted and reused aggressively
153/// without dependency edges on the creating query. When a tracked function is collected,
154/// any tracked structs it created can be deleted. Additionally, when a tracked function
155/// re-executes but does not create a tracked struct that was previously created, it can
156/// be deleted. No dependency edge is required as the lifetime of a tracked struct is tied
157/// directly to the query that created it.

159where
160    C: Configuration,

162    /// Our index in the database.
163    ingredient_index: IngredientIndex,
164
165    /// Phantom data: we fetch `Value<C>` out from `Table`
166    phantom: PhantomData<fn() -> Value<C>>,
167
168    /// Store freed ids
169    free_list: SegQueue<Id>,
170
171    memo_table_types: Arc<MemoTableTypes>,
172}

174/// Defines the identity of a tracked struct.
175/// This is the key to a hashmap that is (initially)
176/// stored in the [`ActiveQuery`](`crate::active_query::ActiveQuery`)
177/// struct and later moved to the [`Memo`](`crate::function::memo::Memo`).
178#[derive(Debug, PartialEq, Eq, PartialOrd, Ord, Copy, Clone)]
179pub(crate) struct Identity {
180    // Conceptually, this contains an `IdentityHash`, but using `IdentityHash` directly will grow the size
181    // of this struct struct by a `std::mem::size_of::<usize>()` due to unusable padding. To avoid this increase
182    // in size, we inline the fields of `IdentityHash`.
183    /// Index of the tracked struct ingredient.
184    ingredient_index: IngredientIndex,
185
186    /// Hash of the id fields.
187    hash: u64,
188
189    /// The unique disambiguator assigned within the active query
190    /// to distinguish distinct tracked structs with the same identity_hash.
191    disambiguator: Disambiguator,
192}
193
194impl Identity {
195    pub(crate) fn ingredient_index(&self) -> IngredientIndex {
196        self.ingredient_index
197    }
198}

200/// Stores the data that (almost) uniquely identifies a tracked struct.
201/// This includes the ingredient index of that struct type plus the hash of its untracked fields.
202/// This is mapped to a disambiguator -- a value that starts as 0 but increments each round,
203/// allowing for multiple tracked structs with the same hash and ingredient_index
204/// created within the query to each have a unique id.
205#[derive(Debug, PartialEq, Eq, PartialOrd, Ord, Copy, Clone)]
206pub struct IdentityHash {
207    /// Index of the tracked struct ingredient.
208    ingredient_index: IngredientIndex,
209
210    /// Hash of the id fields.
211    hash: u64,
212}

214/// A map from tracked struct keys (which include the hash + [Disambiguator]) to their
215/// final [Id].
216#[derive(Default, Debug)]
217pub(crate) struct IdentityMap {
218    // we use a hashtable here as our key contains its own hash (`Identity::hash`)
219    // so we do the hash wrangling ourselves
220    table: hashbrown::HashTable<(Identity, Id)>,
221}
222
223impl Clone for IdentityMap {
224    fn clone(&self) -> Self {
225        Self {
226            table: self.table.clone(),
227        }
228    }
229}
230
231impl IdentityMap {
232    pub(crate) fn clone_from_slice(&mut self, source: &[(Identity, Id)]) {
233        self.table.clear();
234        self.table.reserve(source.len(), |(k, _)| k.hash);
235
236        for (key, id) in source {
237            self.insert(*key, *id);
238        }
239    }
240
241    pub(crate) fn insert(&mut self, key: Identity, id: Id) -> Option<Id> {
242        let entry = self.table.find_mut(key.hash, |&(k, _)| k == key);
243        match entry {
244            Some(occupied) => Some(mem::replace(&mut occupied.1, id)),
245            None => {
246                self.table
247                    .insert_unique(key.hash, (key, id), |(k, _)| k.hash);
248                None
249            }
250        }
251    }
252
253    pub(crate) fn get(&self, key: &Identity) -> Option<Id> {
254        self.table
255            .find(key.hash, |&(k, _)| k == *key)
256            .map(|&(_, v)| v)
257    }
258
259    pub(crate) fn is_empty(&self) -> bool {
260        self.table.is_empty()
261    }
262
263    pub(crate) fn clear(&mut self) {
264        self.table.clear()
265    }
266
267    pub(crate) fn into_thin_vec(self) -> ThinVec<(Identity, Id)> {
268        self.table.into_iter().collect()
269    }
270}

273#[derive(Debug)]
274pub struct Value<C>
275where
276    C: Configuration,
277{
278    /// The minimum durability of all inputs consumed by the creator
279    /// query prior to creating this tracked struct. If any of those
280    /// inputs changes, then the creator query may create this struct
281    /// with different values.
282    durability: Durability,
283
284    /// The revision when this tracked struct was last updated.
285    /// This field also acts as a kind of "lock". Once it is equal
286    /// to `Some(current_revision)`, the fields are locked and
287    /// cannot change further. This makes it safe to give out `&`-references
288    /// so long as they do not live longer than the current revision
289    /// (which is assured by tying their lifetime to the lifetime of an `&`-ref
290    /// to the database).
291    ///
292    /// The struct is updated from an older revision `R0` to the current revision `R1`
293    /// when the struct is first accessed in `R1`, whether that be because the original
294    /// query re-created the struct (i.e., by user calling `Struct::new`) or because
295    /// the struct was read from. (Structs may not be recreated in the new revision if
296    /// the inputs to the query have not changed.)
297    ///
298    /// When re-creating the struct, the field is temporarily set to `None`.
299    /// This is signal that there is an active `&mut` modifying the other fields:
300    /// even reading from those fields in that situation would create UB.
301    /// This `None` value should never be observable by users unless they have
302    /// leaked a reference across threads somehow.
303    updated_at: OptionalAtomicRevision,
304
305    /// Fields of this tracked struct. They can change across revisions,
306    /// but they do not change within a particular revision.
307    fields: C::Fields<'static>,
308
309    /// The revision information for each field: when did this field last change.
310    /// When tracked structs are re-created, this revision may be updated to the
311    /// current revision if the value is different.
312    revisions: C::Revisions,
313
314    /// Memo table storing the results of query functions etc.
315    /*unsafe */
316    memos: MemoTable,
317}

320#[derive(Debug, PartialEq, Eq, PartialOrd, Ord, Copy, Clone)]
321pub struct Disambiguator(u32);
322
323#[derive(Default, Debug)]
324pub(crate) struct DisambiguatorMap {
325    // we use a non-hasher hashmap here as our key contains its own hash (in a sense)
326    // so we use the raw entry api instead to avoid the overhead of hashing unnecessarily
327    map: hashbrown::HashMap<IdentityHash, Disambiguator, ()>,
328}
329
330impl DisambiguatorMap {
331    pub(crate) fn disambiguate(&mut self, key: IdentityHash) -> Disambiguator {
332        use hashbrown::hash_map::RawEntryMut;
333
334        let entry = self.map.raw_entry_mut().from_hash(key.hash, |k| *k == key);
335        let disambiguator = match entry {
336            RawEntryMut::Occupied(occupied) => occupied.into_mut(),
337            RawEntryMut::Vacant(vacant) => {
338                vacant
339                    .insert_with_hasher(key.hash, key, Disambiguator(0), |k| k.hash)
340                    .1
341            }
342        };
343        let result = *disambiguator;
344        disambiguator.0 += 1;
345        result
346    }
347
348    pub fn clear(&mut self) {
349        self.map.clear()
350    }
351
352    pub fn is_empty(&self) -> bool {
353        self.map.is_empty()
354    }
355}

357impl<C> IngredientImpl<C>
358where
359    C: Configuration,
360{
361    /// Create a tracked struct ingredient. Generated by the `#[tracked]` macro,
362    /// not meant to be called directly by end-users.
363    fn new(index: IngredientIndex) -> Self {
364        Self {
365            ingredient_index: index,
366            phantom: PhantomData,
367            free_list: Default::default(),
368            memo_table_types: Arc::new(MemoTableTypes::default()),
369        }
370    }
371
372    /// Returns the database key index for a tracked struct with the given id.
373    pub fn database_key_index(&self, id: Id) -> DatabaseKeyIndex {
374        DatabaseKeyIndex::new(self.ingredient_index, id)
375    }
376
377    pub fn new_struct<'db>(
378        &'db self,
379        db: &'db dyn Database,
380        mut fields: C::Fields<'db>,
381    ) -> C::Struct<'db> {
382        let (zalsa, zalsa_local) = db.zalsas();
383
384        let identity_hash = IdentityHash {
385            ingredient_index: self.ingredient_index,
386            hash: crate::hash::hash(&C::untracked_fields(&fields)),
387        };
388
389        let (current_deps, disambiguator) = zalsa_local.disambiguate(identity_hash);
390
391        let identity = Identity {
392            hash: identity_hash.hash,
393            ingredient_index: identity_hash.ingredient_index,
394            disambiguator,
395        };
396
397        let current_revision = zalsa.current_revision();
398        if let Some(id) = zalsa_local.tracked_struct_id(&identity) {
399            // The struct already exists in the intern map.
400            let index = self.database_key_index(id);
401            tracing::trace!("Reuse tracked struct {id:?}", id = index);
402            zalsa_local.add_output(index);
403
404            // SAFETY: The `id` was present in the interned map, so the value must be initialized.
405            let update_result =
406                unsafe { self.update(zalsa, current_revision, id, &current_deps, fields) };
407
408            fields = match update_result {
409                // Overwrite the previous ID if we are reusing the old slot with new fields.
410                Ok(updated_id) if updated_id != id => {
411                    zalsa_local.store_tracked_struct_id(identity, updated_id);
412                    return FromId::from_id(updated_id);
413                }
414
415                // The id has not changed.
416                Ok(id) => return FromId::from_id(id),
417
418                // Failed to perform the update, we are forced to allocate a new slot.
419                Err(fields) => fields,
420            };
421        }
422
423        // We failed to perform the update, or this is a new tracked struct, so allocate a new entry
424        // in the struct map.
425        let id = self.allocate(zalsa, zalsa_local, current_revision, &current_deps, fields);
426        let key = self.database_key_index(id);
427        tracing::trace!("Allocated new tracked struct {key:?}");
428        zalsa_local.add_output(key);
429        zalsa_local.store_tracked_struct_id(identity, id);
430        FromId::from_id(id)
431    }
432
433    fn allocate<'db>(
434        &'db self,
435        zalsa: &'db Zalsa,
436        zalsa_local: &'db ZalsaLocal,
437        current_revision: Revision,
438        current_deps: &Stamp,
439        fields: C::Fields<'db>,
440    ) -> Id {
441        let value = |_| Value {
442            updated_at: OptionalAtomicRevision::new(Some(current_revision)),
443            durability: current_deps.durability,
444            // lifetime erase for storage
445            fields: unsafe { mem::transmute::<C::Fields<'db>, C::Fields<'static>>(fields) },
446            revisions: C::new_revisions(current_deps.changed_at),
447            memos: Default::default(),
448        };
449
450        while let Some(id) = self.free_list.pop() {
451            // Increment the ID generation before reusing it, as if we have allocated a new
452            // slot in the table.
453            //
454            // If the generation would overflow, we are forced to leak the slot. Note that this
455            // shouldn't be a problem in general as sufficient bits are reserved for the generation.
456            let Some(id) = id.next_generation() else {
457                tracing::info!(
458                    "leaking tracked struct {:?} due to generation overflow",
459                    self.database_key_index(id)
460                );
461
462                continue;
463            };
464
465            // SAFETY: We just removed `id` from the free-list, so we have exclusive access.
466            let data = unsafe { &mut *Self::data_raw(zalsa.table(), id) };
467
468            assert!(
469                data.updated_at.load().is_none(),
470                "free list entry for `{id:?}` does not have `None` for `updated_at`"
471            );
472
473            // Overwrite the free-list entry. Use `*foo = ` because the entry
474            // has been previously initialized and we want to free the old contents.
475            *data = value(id);
476
477            return id;
478        }
479
480        zalsa_local.allocate::<Value<C>>(zalsa, self.ingredient_index, value)
481    }
482
483    /// Get mutable access to the data for `id` -- this holds a write lock for the duration
484    /// of the returned value.
485    ///
486    /// # Panics
487    ///
488    /// * If the value is not present in the map.
489    /// * If the value is already updated in this revision.
490    ///
491    /// # Safety
492    ///
493    /// The value at the given `id` must be initialized.
494    unsafe fn update<'db>(
495        &'db self,
496        zalsa: &'db Zalsa,
497        current_revision: Revision,
498        mut id: Id,
499        current_deps: &Stamp,
500        fields: C::Fields<'db>,
501    ) -> Result<Id, C::Fields<'db>> {
502        let data_raw = Self::data_raw(zalsa.table(), id);
503
504        // The protocol is:
505        //
506        // * When we begin updating, we store `None` in the `updated_at` field
507        // * When completed, we store `Some(current_revision)` in `updated_at`
508        //
509        // No matter what mischief users get up to, it should be impossible for us to
510        // observe `None` in `updated_at`. The `id` should only be associated with one
511        // query and that query can only be running in one thread at a time.
512        //
513        // We *can* observe `Some(current_revision)` however, which means that this
514        // tracked struct is already updated for this revision in two ways.
515        // In that case we should not modify or touch it because there may be
516        // `&`-references to its contents floating around.
517        //
518        // Observing `Some(current_revision)` can happen in two scenarios: leaks (tsk tsk)
519        // but also the scenario embodied by the test test `test_run_5_then_20` in `specify_tracked_fn_in_rev_1_but_not_2.rs`:
520        //
521        // * Revision 1:
522        //   * Tracked function F creates tracked struct S
523        //   * F reads input I
524        // * Revision 2: I is changed, F is re-executed
525        //
526        // When F is re-executed in rev 2, we first try to validate F's inputs/outputs,
527        // which is the list [output: S, input: I]. As no inputs have changed by the time
528        // we reach S, we mark it as verified. But then input I is seen to have changed,
529        // and so we re-execute F. Note that we *know* that S will have the same value
530        // (barring program bugs).
531        //
532        // Further complicating things: it is possible that F calls F2
533        // and gives it (e.g.) S as one of its arguments. Validating F2 may cause F2 to
534        // re-execute which means that it may indeed have read from S's fields
535        // during the current revision and thus obtained an `&` reference to those fields
536        // that is still live.
537
538        {
539            // SAFETY: Guaranteed by caller.
540            let data = unsafe { &*data_raw };
541
542            let last_updated_at = data.updated_at.load();
543            assert!(
544                last_updated_at.is_some(),
545                "two concurrent writers to {id:?}, should not be possible"
546            );
547
548            // The value is already read-locked, but we can reuse it safely as per above.
549            if last_updated_at == Some(current_revision) {
550                return Ok(id);
551            }
552
553            // Updating the fields may make it necessary to increment the generation of the ID. In
554            // the unlikely case that the ID is already at its maximum generation, we are forced to leak
555            // the previous slot and allocate a new value.
556            if id.generation() == u32::MAX {
557                tracing::info!(
558                    "leaking tracked struct {:?} due to generation overflow",
559                    self.database_key_index(id)
560                );
561
562                return Err(fields);
563            }
564
565            // Acquire the write-lock. This can only fail if there is a parallel thread
566            // reading from this same `id`, which can only happen if the user has leaked it.
567            // Tsk tsk.
568            let swapped_out = data.updated_at.swap(None);
569            if swapped_out != last_updated_at {
570                panic!(
571                "failed to acquire write lock, id `{id:?}` must have been leaked across threads"
572            );
573            }
574        }
575
576        // UNSAFE: Marking as mut requires exclusive access for the duration of
577        // the `mut`. We have now *claimed* this data by swapping in `None`,
578        // any attempt to read concurrently will panic.
579        let data = unsafe { &mut *data_raw };
580
581        // SAFETY: We assert that the pointer to `data.revisions`
582        // is a pointer into the database referencing a value
583        // from a previous revision. As such, it continues to meet
584        // its validity invariant and any owned content also continues
585        // to meet its safety invariant.
586        let untracked_update = unsafe {
587            C::update_fields(
588                current_deps.changed_at,
589                &mut data.revisions,
590                mem::transmute::<*mut C::Fields<'static>, *mut C::Fields<'db>>(
591                    std::ptr::addr_of_mut!(data.fields),
592                ),
593                fields,
594            )
595        };
596
597        if untracked_update {
598            // Consider this a new tracked-struct when any non-tracked field got updated.
599            // This should be rare and only ever happen if there's a hash collision.
600            //
601            // Note that we hold the lock and have exclusive access to the tracked struct data,
602            // so there should be no live instances of IDs from the previous generation. We clear
603            // the memos and return a new ID here as if we have allocated a new slot.
604            let mut table = data.take_memo_table();
605
606            // SAFETY: The memo table belongs to a value that we allocated, so it has the
607            // correct type.
608            unsafe { self.clear_memos(zalsa, &mut table, id) };
609
610            id = id
611                .next_generation()
612                .expect("already verified that generation is not maximum");
613        }
614
615        if current_deps.durability < data.durability {
616            data.revisions = C::new_revisions(current_deps.changed_at);
617        }
618        data.durability = current_deps.durability;
619        let swapped_out = data.updated_at.swap(Some(current_revision));
620        assert!(swapped_out.is_none());
621
622        Ok(id)
623    }

635    /// Deletes the given entities. This is used after a query `Q` executes and we can compare
636    /// the entities `E_now` that it produced in this revision vs the entities
637    /// `E_prev` it produced in the last revision. Any missing entities `E_prev - E_new` can be
638    /// deleted.
639    ///
640    /// # Warning
641    ///
642    /// Using this method on an entity id that MAY be used in the current revision will lead to
643    /// unspecified results (but not UB). See [`InternedIngredient::delete_index`] for more
644    /// discussion and important considerations.
645    pub(crate) fn delete_entity(&self, zalsa: &Zalsa, id: Id) {
646        zalsa.event(&|| {
647            Event::new(crate::EventKind::DidDiscard {
648                key: self.database_key_index(id),
649            })
650        });
651
652        let current_revision = zalsa.current_revision();
653        let data_raw = Self::data_raw(zalsa.table(), id);
654
655        {
656            let data = unsafe { &*data_raw };
657
658            // We want to set `updated_at` to `None`, signalling that other field values
659            // cannot be read. The current value should be `Some(R0)` for some older revision.
660            match data.updated_at.load() {
661                None => {
662                    panic!("cannot delete write-locked id `{id:?}`; value leaked across threads");
663                }
664                Some(r) if r == current_revision => panic!(
665                    "cannot delete read-locked id `{id:?}`; value leaked across threads or user functions not deterministic"
666                ),
667                Some(r) => {
668                    if data.updated_at.compare_exchange(Some(r), None).is_err() {
669                        panic!("race occurred when deleting value `{id:?}`")
670                    }
671                }
672            }
673        }
674
675        // SAFETY: We have acquired the write lock
676        let data = unsafe { &mut *data_raw };
677        let mut memo_table = data.take_memo_table();
678
679        // SAFETY: The memo table belongs to a value that we allocated, so it
680        // has the correct type.
681        unsafe { self.clear_memos(zalsa, &mut memo_table, id) };
682
683        // now that all cleanup has occurred, make available for re-use
684        self.free_list.push(id);
685    }
686
687    /// Clears the given memo table.
688    ///
689    /// # Safety
690    ///
691    /// The `MemoTable` must belong to a `Value` of the correct type.
692    pub(crate) unsafe fn clear_memos(&self, zalsa: &Zalsa, memo_table: &mut MemoTable, id: Id) {
693        // SAFETY: The caller guarantees this is the correct types table.
694        let table = unsafe { self.memo_table_types.attach_memos_mut(memo_table) };
695
696        // `Database::salsa_event` is a user supplied callback which may panic
697        // in that case we need a drop guard to free the memo table
698        struct TableDropGuard<'a>(MemoTableWithTypesMut<'a>);
699        impl Drop for TableDropGuard<'_> {
700            fn drop(&mut self) {
701                // SAFETY: We have `&mut MemoTable`, so no more references to these memos exist and we are good
702                // to drop them.
703                unsafe { self.0.drop() };
704            }
705        }
706
707        let mut table_guard = TableDropGuard(table);
708
709        // SAFETY: We have `&mut MemoTable`, so no more references to these memos exist and we are good
710        // to drop them.
711        unsafe {
712            table_guard.0.take_memos(|memo_ingredient_index, memo| {
713                let ingredient_index =
714                    zalsa.ingredient_index_for_memo(self.ingredient_index, memo_ingredient_index);
715
716                let executor = DatabaseKeyIndex::new(ingredient_index, id);
717
718                zalsa.event(&|| Event::new(EventKind::DidDiscard { key: executor }));
719
720                for stale_output in memo.origin().outputs() {
721                    stale_output.remove_stale_output(zalsa, executor);
722                }
723            })
724        };
725
726        mem::forget(table_guard);
727    }

741    /// Access to this tracked field.
742    ///
743    /// Note that this function returns the entire tuple of value fields.
744    /// The caller is responsible for selecting the appropriate element.
745    pub fn tracked_field<'db>(
746        &'db self,
747        db: &'db dyn crate::Database,
748        s: C::Struct<'db>,
749        relative_tracked_index: usize,
750    ) -> &'db C::Fields<'db> {
751        let (zalsa, zalsa_local) = db.zalsas();
752        let id = AsId::as_id(&s);
753        let field_ingredient_index = self.ingredient_index.successor(relative_tracked_index);
754        let data = Self::data(zalsa.table(), id);
755
756        data.read_lock(zalsa.current_revision());
757
758        let field_changed_at = data.revisions[relative_tracked_index];
759
760        zalsa_local.report_tracked_read_simple(
761            DatabaseKeyIndex::new(field_ingredient_index, id),
762            data.durability,
763            field_changed_at,
764        );
765
766        data.fields()
767    }

769    /// Access to this untracked field.
770    ///
771    /// Note that this function returns the entire tuple of value fields.
772    /// The caller is responsible for selecting the appropriate element.
773    pub fn untracked_field<'db>(
774        &'db self,
775        db: &'db dyn crate::Database,
776        s: C::Struct<'db>,
777    ) -> &'db C::Fields<'db> {
778        let zalsa = db.zalsa();
779        let id = AsId::as_id(&s);
780        let data = Self::data(zalsa.table(), id);
781
782        data.read_lock(zalsa.current_revision());
783
784        // Note that we do not need to add a dependency on the tracked struct
785        // as IDs that are reused increment their generation, invalidating any
786        // dependent queries directly.
787
788        data.fields()
789    }

880impl<C> Value<C>
881where
882    C: Configuration,
883{
884    /// Fields of this tracked struct.
885    ///
886    /// They can change across revisions, but they do not change within
887    /// a particular revision.
888    #[cfg_attr(not(feature = "salsa_unstable"), doc(hidden))]
889    pub fn fields(&self) -> &C::Fields<'_> {
890        // SAFETY: We are shrinking the lifetime from storage back to the db lifetime.
891        unsafe { mem::transmute::<&C::Fields<'static>, &C::Fields<'_>>(&self.fields) }
892    }
893
894    fn take_memo_table(&mut self) -> MemoTable {
895        // This fn is only called after `updated_at` has been set to `None`;
896        // this ensures that there is no concurrent access
897        // (and that the `&mut self` is accurate...).
898        assert!(self.updated_at.load().is_none());
899
900        mem::take(&mut self.memos)
901    }
902
903    fn read_lock(&self, current_revision: Revision) {
904        loop {
905            match self.updated_at.load() {
906                None => {
907                    panic!("access to field whilst the value is being initialized");
908                }
909                Some(r) => {
910                    if r == current_revision {
911                        return;
912                    }
913
914                    if self
915                        .updated_at
916                        .compare_exchange(Some(r), Some(current_revision))
917                        .is_ok()
918                    {
919                        break;
920                    }
921                }
922            }
923        }
924    }
925
926    /// Returns memory usage information about the tracked struct.
927    ///
928    /// # Safety
929    ///
930    /// The `MemoTable` must belong to a `Value` of the correct type.
931    #[cfg(feature = "salsa_unstable")]
932    unsafe fn memory_usage(&self, memo_table_types: &MemoTableTypes) -> crate::database::SlotInfo {
933        // SAFETY: The caller guarantees this is the correct types table.
934        let memos = unsafe { memo_table_types.attach_memos(&self.memos) };
935
936        crate::database::SlotInfo {
937            debug_name: C::DEBUG_NAME,
938            size_of_metadata: mem::size_of::<Self>() - mem::size_of::<C::Fields<'_>>(),
939            size_of_fields: mem::size_of::<C::Fields<'_>>(),
940            memos: memos.memory_usage(),
941        }
942    }
943}

945// SAFETY: `Value<C>` is our private type branded over the unique configuration `C`.
946unsafe impl<C> Slot for Value<C>
947where
948    C: Configuration,
949{
950    #[inline(always)]
951    unsafe fn memos(&self, current_revision: Revision) -> &crate::table::memo::MemoTable {
952        // Acquiring the read lock here with the current revision
953        // ensures that there is no danger of a race
954        // when deleting a tracked struct.
955        self.read_lock(current_revision);
956        &self.memos
957    }
958
959    #[inline(always)]
960    fn memos_mut(&mut self) -> &mut crate::table::memo::MemoTable {
961        &mut self.memos
962    }
