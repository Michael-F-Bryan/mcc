26/// Trait that defines the key properties of an interned struct.
27///
28/// Implemented by the `#[salsa::interned]` macro when applied to
29/// a struct.
30pub trait Configuration: Sized + 'static {
31    const LOCATION: crate::ingredient::Location;
32
33    const DEBUG_NAME: &'static str;
34
35    // The minimum number of revisions that must pass before a stale value is garbage collected.
36    #[cfg(test)]
37    const REVISIONS: NonZeroUsize = NonZeroUsize::new(3).unwrap();
38
39    #[cfg(not(test))] // More aggressive garbage collection by default when testing.
40    const REVISIONS: NonZeroUsize = NonZeroUsize::new(1).unwrap();

42    /// The fields of the struct being interned.
43    type Fields<'db>: InternedData;
44
45    /// The end user struct

56/// The interned ingredient hashes values of type `C::Fields` to produce an `Id`.
57///
58/// It used to store interned structs but also to store the ID fields of a tracked struct.
59/// Interned values are garbage collected and their memory reused based on an LRU heuristic.
60pub struct IngredientImpl<C: Configuration> {
61    /// Index of this ingredient in the database (used to construct database-IDs, etc).
62    ingredient_index: IngredientIndex,
63
64    /// A hasher for the sharded ID maps.
65    hasher: FxBuildHasher,
66
67    /// A shift used to determine the shard for a given hash.
68    shift: u32,
69
70    /// Sharded data that can only be accessed through a lock.
71    shards: Box<[CachePadded<Mutex<IngredientShard<C>>>]>,
72
73    /// A queue of recent revisions in which values were interned.
74    revision_queue: RevisionQueue<C>,
75
76    memo_table_types: Arc<MemoTableTypes>,
77
78    _marker: PhantomData<fn() -> C>,
79}

81struct IngredientShard<C: Configuration> {

83    ///
84    /// This doesn't hold the fields themselves to save memory, instead it points

135/// Shared value data can only be read through the lock.
136#[repr(Rust, packed)] // Allow `durability` to be stored in the padding of the outer `Value` struct.
137struct ValueShared {
138    /// The interned ID for this value.
139    ///
140    /// Storing this on the value itself is necessary to identify slots
141    /// from the LRU list, as well as keep track of the generation.
142    ///
143    /// Values that are reused increment the ID generation, as if they had
144    /// allocated a new slot. This eliminates the need for dependency edges
145    /// on queries that *read* from an interned value, as any memos dependent
146    /// on the previous value will not match the new ID.
147    ///
148    /// However, reusing a slot invalidates the previous ID, so dependency edges
149    /// on queries that *create* an interned value are still required to ensure
150    /// the value is re-interned with a new ID.
151    id: Id,
152
153    /// The revision the value was most-recently interned in.
154    last_interned_at: Revision,
155
156    /// The minimum durability of all inputs consumed by the creator
157    /// query prior to creating this interned struct. If any of those
158    /// inputs changes, then the creator query may create this struct
159    /// with different values.
160    durability: Durability,
161}
162
163impl ValueShared {
164    /// Returns `true` if this value slot can be reused when interning, and should be added to the LRU.
165    fn is_reusable<C: Configuration>(&self) -> bool {
166        // Garbage collection is disabled.
167        if C::REVISIONS == IMMORTAL {
168            return false;
169        }
170
171        // Collecting higher durability values requires invalidating the revision for their
172        // durability (see `Database::synthetic_write`, which requires a mutable reference to
173        // the database) to avoid short-circuiting calls to `maybe_changed_after`. This is
174        // necessary because `maybe_changed_after` for interned values is not "pure"; it updates
175        // the `last_interned_at` field before validating a given value to ensure that it is not
176        // reused after read in the current revision.
177        self.durability == Durability::LOW
178    }
179}

194    /// Returns memory usage information about the interned value.
195    ///
196    /// # Safety
197    ///
198    /// The `MemoTable` must belong to a `Value` of the correct type. Additionally, the
199    /// lock must be held for the shard containing the value.
200    #[cfg(all(not(feature = "shuttle"), feature = "salsa_unstable"))]
201    unsafe fn memory_usage(&self, memo_table_types: &MemoTableTypes) -> crate::database::SlotInfo {
202        // SAFETY: The caller guarantees we hold the lock for the shard containing the value, so we
203        // have at-least read-only access to the value's memos.
204        let memos = unsafe { &*self.memos.get() };
205        // SAFETY: The caller guarantees this is the correct types table.
206        let memos = unsafe { memo_table_types.attach_memos(memos) };
207
208        crate::database::SlotInfo {
209            debug_name: C::DEBUG_NAME,
210            size_of_metadata: std::mem::size_of::<Self>() - std::mem::size_of::<C::Fields<'_>>(),
211            size_of_fields: std::mem::size_of::<C::Fields<'_>>(),
212            memos: memos.memory_usage(),
213        }
214    }

243    pub fn new(ingredient_index: IngredientIndex) -> Self {
244        static SHARDS: OnceLock<usize> = OnceLock::new();
245        let shards = *SHARDS.get_or_init(|| {
246            let num_cpus = std::thread::available_parallelism()
247                .map(usize::from)
248                .unwrap_or(1);
249
250            (num_cpus * 4).next_power_of_two()
251        });
252
253        Self {
254            ingredient_index,
255            hasher: FxBuildHasher,
256            memo_table_types: Arc::new(MemoTableTypes::default()),
257            revision_queue: RevisionQueue::default(),
258            shift: usize::BITS - shards.trailing_zeros(),
259            shards: (0..shards).map(|_| Default::default()).collect(),
260            _marker: PhantomData,
261        }
262    }

264    /// Returns the shard for a given hash.
265    ///
266    /// Note that this value is guaranteed to be in-bounds for `self.shards`.
267    #[inline]
268    fn shard(&self, hash: u64) -> usize {
269        // https://github.com/xacrimon/dashmap/blob/366ce7e7872866a06de66eb95002fa6cf2c117a7/src/lib.rs#L421
270        ((hash as usize) << 7) >> self.shift
271    }

311    /// Intern data to a unique reference.
312    ///
313    /// If `key` is already interned, returns the existing [`Id`] for the interned data without
314    /// invoking `assemble`.
315    ///
316    /// Otherwise, invokes `assemble` with the given `key` and the [`Id`] to be allocated for this
317    /// interned value. The resulting [`C::Data`] will then be interned.
318    ///
319    /// Note: Using the database within the `assemble` function may result in a deadlock if
320    /// the database ends up trying to intern or allocate a new value.
321    pub fn intern_id<'db, Key>(
322        &'db self,
323        db: &'db dyn crate::Database,
324        key: Key,
325        assemble: impl FnOnce(Id, Key) -> C::Fields<'db>,
326    ) -> crate::Id
327    where
328        Key: Hash,
329        // We'd want the following predicate, but this currently implies `'static` due to a rustc
330        // bug
331        // for<'db> C::Data<'db>: HashEqLike<Key>,
332        // so instead we go with this and transmute the lifetime in the `eq` closure
333        C::Fields<'db>: HashEqLike<Key>,
334    {
335        let (zalsa, zalsa_local) = db.zalsas();
336
337        // Record the current revision as active.
338        let current_revision = zalsa.current_revision();
339        self.revision_queue.record(current_revision);
340
341        // Hash the value before acquiring the lock.
342        let hash = self.hasher.hash_one(&key);
343
344        let shard_index = self.shard(hash);
345        // SAFETY: `shard_index` is guaranteed to be in-bounds for `self.shards`.
346        let shard = unsafe { &mut *self.shards.get_unchecked(shard_index).lock() };
347
348        let found_value = Cell::new(None);
349        // SAFETY: We hold the lock for the shard containing the value.
350        let eq = |id: &_| unsafe { Self::value_eq(*id, &key, zalsa, &found_value) };
351
352        // Attempt a fast-path lookup of already interned data.
353        if let Some(&id) = shard.key_map.find(hash, eq) {
354            let value = found_value
355                .get()
356                .expect("found the interned value, so `found_value` should be set");
357
358            let index = self.database_key_index(id);
359
360            // SAFETY: We hold the lock for the shard containing the value.
361            let value_shared = unsafe { &mut *value.shared.get() };
362
363            // Validate the value in this revision to avoid reuse.
364            if { value_shared.last_interned_at } < current_revision {
365                value_shared.last_interned_at = current_revision;
366
367                zalsa.event(&|| {
368                    Event::new(EventKind::DidValidateInternedValue {
369                        key: index,
370                        revision: current_revision,
371                    })
372                });
373
374                if value_shared.is_reusable::<C>() {
375                    // Move the value to the front of the LRU list.
376                    //
377                    // SAFETY: We hold the lock for the shard containing the value, and `value` is
378                    // a reusable value that was previously interned, so is in the list.
379                    unsafe { shard.lru.cursor_mut_from_ptr(value).remove() };
380
381                    // SAFETY: The value pointer is valid for the lifetime of the database
382                    // and never accessed mutably directly.
383                    unsafe { shard.lru.push_front(UnsafeRef::from_raw(value)) };
384                }
385            }
386
387            if let Some((_, stamp)) = zalsa_local.active_query() {
388                let was_reusable = value_shared.is_reusable::<C>();
389
390                // Record the maximum durability across all queries that intern this value.
391                value_shared.durability = std::cmp::max(value_shared.durability, stamp.durability);
392
393                // If the value is no longer reusable, i.e. the durability increased, remove it
394                // from the LRU.
395                if was_reusable && !value_shared.is_reusable::<C>() {
396                    // SAFETY: We hold the lock for the shard containing the value, and `value`
397                    // was previously reusable, so is in the list.
398                    unsafe { shard.lru.cursor_mut_from_ptr(value).remove() };
399                }
400            }
401
402            // Record a dependency on the value.
403            //
404            // See `intern_id_cold` for why we need to use `current_revision` here. Note that just
405            // because this value was previously interned does not mean it was previously interned
406            // by *our query*, so the same considerations apply.
407            zalsa_local.report_tracked_read_simple(
408                index,
409                value_shared.durability,
410                current_revision,
411            );
412
413            return value_shared.id;
414        }
415
416        // Fill up the table for the first few revisions without attempting garbage collection.
417        if !self.revision_queue.is_primed() {
418            return self.intern_id_cold(
419                db,
420                key,
421                zalsa,
422                zalsa_local,
423                assemble,
424                shard,
425                shard_index,
426                hash,
427            );
428        }
429
430        // Otherwise, try to reuse a stale slot.
431        let mut cursor = shard.lru.back_mut();
432
433        while let Some(value) = cursor.get() {
434            // SAFETY: We hold the lock for the shard containing the value.
435            let value_shared = unsafe { &mut *value.shared.get() };
436
437            // The value must not have been read in the current revision to be collected
438            // soundly, but we also do not want to collect values that have been read recently.
439            //
440            // Note that the list is sorted by LRU, so if the tail of the list is not stale, we
441            // will not find any stale slots.
442            if !self.revision_queue.is_stale(value_shared.last_interned_at) {
443                break;
444            }
445
446            // We should never reuse a value that was accessed in the current revision.
447            debug_assert!({ value_shared.last_interned_at } < current_revision);
448
449            // Record the durability of the current query on the interned value.
450            let (durability, last_interned_at) = zalsa_local
451                .active_query()
452                .map(|(_, stamp)| (stamp.durability, current_revision))
453                // If there is no active query this durability does not actually matter.
454                // `last_interned_at` needs to be `Revision::MAX`, see the `intern_access_in_different_revision` test.
455                .unwrap_or((Durability::MAX, Revision::max()));
456
457            let old_id = value_shared.id;
458
459            // Increment the generation of the ID, as if we allocated a new slot.
460            //
461            // If the ID is at its maximum generation, we are forced to leak the slot.
462            let Some(new_id) = value_shared.id.next_generation() else {
463                // Remove the value from the LRU list as we will never be able to
464                // collect it.
465                cursor.remove().unwrap();
466
467                // Retry with the previous element.
468                cursor = shard.lru.back_mut();
469
470                continue;
471            };
472
473            // Mark the slot as reused.
474            *value_shared = ValueShared {
475                id: new_id,
476                durability,
477                last_interned_at,
478            };
479
480            let index = self.database_key_index(value_shared.id);
481
482            // Record a dependency on the new value.
483            //
484            // See `intern_id_cold` for why we need to use `current_revision` here.
485            zalsa_local.report_tracked_read_simple(
486                index,
487                value_shared.durability,
488                current_revision,
489            );
490
491            zalsa.event(&|| {
492                Event::new(EventKind::DidReuseInternedValue {
493                    key: index,
494                    revision: current_revision,
495                })
496            });
497
498            // Remove the value from the LRU list.
499            //
500            // SAFETY: The value pointer is valid for the lifetime of the database.
501            let value = unsafe { &*UnsafeRef::into_raw(cursor.remove().unwrap()) };
502
503            // SAFETY: We hold the lock for the shard containing the value, and the
504            // value has not been interned in the current revision, so no references to
505            // it can exist.
506            let old_fields = unsafe { &mut *value.fields.get() };
507
508            // Remove the previous value from the ID map.
509            //
510            // Note that while the ID stays the same when a slot is reused, the fields,
511            // and thus the hash, will change, so we need to re-insert the value into the
512            // map. Crucially, we know that the hashes for the old and new fields both map
513            // to the same shard, because we determined the initial shard based on the new
514            // fields and only accessed the LRU list for that shard.
515            let old_hash = self.hasher.hash_one(&*old_fields);
516            shard
517                .key_map
518                .find_entry(old_hash, |found_id: &Id| *found_id == old_id)
519                .expect("interned value in LRU so must be in key_map")
520                .remove();
521
522            // Update the fields.
523            //
524            // SAFETY: We call `from_internal_data` to restore the correct lifetime before access.
525            *old_fields = unsafe { self.to_internal_data(assemble(new_id, key)) };
526
527            // SAFETY: We hold the lock for the shard containing the value.
528            let hasher = |id: &_| unsafe { self.value_hash(*id, zalsa) };
529
530            // Insert the new value into the ID map.
531            shard.key_map.insert_unique(hash, new_id, hasher);
532
533            // Free the memos associated with the previous interned value.
534            //
535            // SAFETY: We hold the lock for the shard containing the value, and the
536            // value has not been interned in the current revision, so no references to
537            // it can exist.
538            let mut memo_table = unsafe { std::mem::take(&mut *value.memos.get()) };
539
540            // SAFETY: The memo table belongs to a value that we allocated, so it has the
541            // correct type.
542            unsafe { self.clear_memos(zalsa, &mut memo_table, new_id) };
543
544            if value_shared.is_reusable::<C>() {
545                // Move the value to the front of the LRU list.
546                //
547                // SAFETY: The value pointer is valid for the lifetime of the database.
548                // and never accessed mutably directly.
549                shard.lru.push_front(unsafe { UnsafeRef::from_raw(value) });
550            }
551
552            return new_id;
553        }
554
555        // If we could not find any stale slots, we are forced to allocate a new one.
556        self.intern_id_cold(
557            db,
558            key,
559            zalsa,
560            zalsa_local,
561            assemble,
562            shard,
563            shard_index,
564            hash,
565        )
566    }

568    /// The cold path for interning a value, allocating a new slot.
569    ///
570    /// Returns `true` if the current thread interned the value.
571    #[allow(clippy::too_many_arguments)]
572    fn intern_id_cold<'db, Key>(
573        &'db self,
574        _db: &'db dyn crate::Database,
575        key: Key,
576        zalsa: &Zalsa,
577        zalsa_local: &ZalsaLocal,
578        assemble: impl FnOnce(Id, Key) -> C::Fields<'db>,
579        shard: &mut IngredientShard<C>,
580        shard_index: usize,
581        hash: u64,
582    ) -> crate::Id
583    where
584        Key: Hash,
585        C::Fields<'db>: HashEqLike<Key>,
586    {
587        let current_revision = zalsa.current_revision();
588
589        // Record the durability of the current query on the interned value.
590        let (durability, last_interned_at) = zalsa_local
591            .active_query()
592            .map(|(_, stamp)| (stamp.durability, current_revision))
593            // If there is no active query this durability does not actually matter.
594            // `last_interned_at` needs to be `Revision::MAX`, see the `intern_access_in_different_revision` test.
595            .unwrap_or((Durability::MAX, Revision::max()));
596
597        // Allocate the value slot.
598        let id = zalsa_local.allocate(zalsa, self.ingredient_index, |id| Value::<C> {
599            shard: shard_index as u16,
600            link: LinkedListLink::new(),
601            memos: UnsafeCell::new(MemoTable::default()),
602            // SAFETY: We call `from_internal_data` to restore the correct lifetime before access.
603            fields: UnsafeCell::new(unsafe { self.to_internal_data(assemble(id, key)) }),
604            shared: UnsafeCell::new(ValueShared {
605                id,
606                durability,
607                last_interned_at,
608            }),
609        });
610
611        let value = zalsa.table().get::<Value<C>>(id);
612        // SAFETY: We hold the lock for the shard containing the value.
613        let value_shared = unsafe { &mut *value.shared.get() };
614
615        if value_shared.is_reusable::<C>() {
616            // Add the value to the front of the LRU list.
617            //
618            // SAFETY: The value pointer is valid for the lifetime of the database
619            // and never accessed mutably directly.
620            shard.lru.push_front(unsafe { UnsafeRef::from_raw(value) });
621        }
622
623        // SAFETY: We hold the lock for the shard containing the value.
624        let hasher = |id: &_| unsafe { self.value_hash(*id, zalsa) };
625
626        // Insert the value into the ID map.
627        shard.key_map.insert_unique(hash, id, hasher);
628
629        debug_assert_eq!(hash, {
630            let value = zalsa.table().get::<Value<C>>(id);
631
632            // SAFETY: We hold the lock for the shard containing the value.
633            unsafe { self.hasher.hash_one(&*value.fields.get()) }
634        });
635
636        let index = self.database_key_index(id);
637
638        // Record a dependency on the newly interned value.
639        //
640        // Note that the ID is unique to this use of the interned slot, so it seems logical to use
641        // `Revision::start()` here. However, it is possible that the ID we read is different from
642        // the previous execution of this query if the previous slot has been reused. In that case,
643        // the query has changed without a corresponding input changing. Using `current_revision`
644        // for dependencies on interned values encodes the fact that interned IDs are not stable
645        // across revisions.
646        zalsa_local.report_tracked_read_simple(index, durability, current_revision);
647
648        zalsa.event(&|| {
649            Event::new(EventKind::DidInternValue {
650                key: index,
651                revision: current_revision,
652            })
653        });
654
655        id
656    }

658    /// Clears the given memo table.
659    ///
660    /// # Safety
661    ///
662    /// The `MemoTable` must belong to a `Value` of the correct type.
663    pub(crate) unsafe fn clear_memos(&self, zalsa: &Zalsa, memo_table: &mut MemoTable, id: Id) {
664        // SAFETY: The caller guarantees this is the correct types table.
665        let table = unsafe { self.memo_table_types.attach_memos_mut(memo_table) };
666
667        // `Database::salsa_event` is a user supplied callback which may panic
668        // in that case we need a drop guard to free the memo table
669        struct TableDropGuard<'a>(MemoTableWithTypesMut<'a>);
670
671        impl Drop for TableDropGuard<'_> {
672            fn drop(&mut self) {
673                // SAFETY: We have `&mut MemoTable`, so no more references to these memos exist and we are good
674                // to drop them.
675                unsafe { self.0.drop() };
676            }
677        }
678
679        let mut table_guard = TableDropGuard(table);
680
681        // SAFETY: We have `&mut MemoTable`, so no more references to these memos exist and we are good
682        // to drop them.
683        unsafe {
684            table_guard.0.take_memos(|memo_ingredient_index, memo| {
685                let ingredient_index =
686                    zalsa.ingredient_index_for_memo(self.ingredient_index, memo_ingredient_index);
687
688                let executor = DatabaseKeyIndex::new(ingredient_index, id);
689
690                zalsa.event(&|| Event::new(EventKind::DidDiscard { key: executor }));
691
692                for stale_output in memo.origin().outputs() {
693                    stale_output.remove_stale_output(zalsa, executor);
694                }
695            })
696        };
697
698        std::mem::forget(table_guard);
699    }

701    // Hashes the value by its fields.
702    //
703    // # Safety
704    //
705    // The lock must be held for the shard containing the value.
706    unsafe fn value_hash<'db>(&'db self, id: Id, zalsa: &'db Zalsa) -> u64 {
707        // This closure is only called if the table is resized. So while it's expensive
708        // to lookup all values, it will only happen rarely.
709        let value = zalsa.table().get::<Value<C>>(id);
710
711        // SAFETY: We hold the lock for the shard containing the value.
712        unsafe { self.hasher.hash_one(&*value.fields.get()) }
713    }

715    // Compares the value by its fields to the given key.
716    //
717    // # Safety
718    //
719    // The lock must be held for the shard containing the value.
720    unsafe fn value_eq<'db, Key>(
721        id: Id,
722        key: &Key,
723        zalsa: &'db Zalsa,
724        found_value: &Cell<Option<&'db Value<C>>>,
725    ) -> bool
726    where
727        C::Fields<'db>: HashEqLike<Key>,
728    {
729        let value = zalsa.table().get::<Value<C>>(id);
730        found_value.set(Some(value));
731
732        // SAFETY: We hold the lock for the shard containing the value.
733        let fields = unsafe { &*value.fields.get() };
734
735        HashEqLike::eq(Self::from_internal_data(fields), key)
736    }

738    /// Returns the database key index for an interned value with the given id.
739    #[inline]
740    pub fn database_key_index(&self, id: Id) -> DatabaseKeyIndex {
741        DatabaseKeyIndex::new(self.ingredient_index, id)
742    }

744    /// Lookup the data for an interned value based on its ID.
745    pub fn data<'db>(&'db self, db: &'db dyn Database, id: Id) -> &'db C::Fields<'db> {
746        let zalsa = db.zalsa();
747        let value = zalsa.table().get::<Value<C>>(id);
748
749        debug_assert!(
750            {
751                let _shard = self.shards[value.shard as usize].lock();
752
753                // SAFETY: We hold the lock for the shard containing the value.
754                let value_shared = unsafe { &mut *value.shared.get() };
755
756                let last_changed_revision = zalsa.last_changed_revision(value_shared.durability);
757                ({ value_shared.last_interned_at }) >= last_changed_revision
758            },
759            "Data was not interned in the latest revision for its durability."
760        );
761
762        // SAFETY: Interned values are only exposed if they have been validated in the
763        // current revision, as checked by the assertion above, which ensures that they
764        // are not reused while being accessed.
765        unsafe { Self::from_internal_data(&*value.fields.get()) }
766    }

769    ///
770    /// Note that this is not "leaking" since no dependency edge is required.
771    pub fn fields<'db>(&'db self, db: &'db dyn Database, s: C::Struct<'db>) -> &'db C::Fields<'db> {
772        self.data(db, AsId::as_id(&s))
773    }

775    pub fn reset(&mut self, db: &mut dyn Database) {
776        _ = db.zalsa_mut();
777
778        for shard in self.shards.iter() {
779            // We can clear the key maps now that we have cancelled all other handles.
780            shard.lock().key_map.clear();
781        }
782    }

807        &self,
808        db: &dyn Database,
809        input: Id,
810        _revision: Revision,
811        _cycle_heads: &mut CycleHeads,
812    ) -> VerifyResult {
813        let zalsa = db.zalsa();
814
815        // Record the current revision as active.
816        let current_revision = zalsa.current_revision();
817        self.revision_queue.record(current_revision);
818
819        let value = zalsa.table().get::<Value<C>>(input);
820
821        // SAFETY: `value.shard` is guaranteed to be in-bounds for `self.shards`.
822        let _shard = unsafe { self.shards.get_unchecked(value.shard as usize) }.lock();
823
824        // SAFETY: We hold the lock for the shard containing the value.
825        let value_shared = unsafe { &mut *value.shared.get() };
826
827        // The slot was reused.
828        if value_shared.id.generation() > input.generation() {
829            return VerifyResult::Changed;
830        }
831
832        // Validate the value for the current revision to avoid reuse.
833        value_shared.last_interned_at = current_revision;
834
835        zalsa.event(&|| {
836            let index = self.database_key_index(input);
837
838            Event::new(EventKind::DidValidateInternedValue {
839                key: index,
840                revision: current_revision,
841            })
842        });
843
844        // Any change to an interned value results in a new ID generation.
845        VerifyResult::unchanged()
846    }

859        use parking_lot::lock_api::RawMutex;
860
861        for shard in self.shards.iter() {
862            // SAFETY: We do not hold any active mutex guards.
863            unsafe { shard.raw().lock() };
864        }
865
866        let memory_usage = self
867            .entries(db)
868            // SAFETY: The memo table belongs to a value that we allocated, so it
869            // has the correct type. Additionally, we are holding the locks for all shards.
870            .map(|value| unsafe { value.memory_usage(&self.memo_table_types) })
871            .collect();
872
873        for shard in self.shards.iter() {
874            // SAFETY: We acquired the locks for all shards.
875            unsafe { shard.raw().unlock() };
876        }
877
878        Some(memory_usage)
879    }

911/// Keep track of revisions in which interned values were read, to determine staleness.
912///
913/// An interned value is considered stale if it has not been read in the past `REVS`
914/// revisions. However, we only consider revisions in which interned values were actually
915/// read, as revisions may be created in bursts.
916struct RevisionQueue<C> {
917    lock: Mutex<()>,
918    // Once `feature(generic_const_exprs)` is stable this can just be an array.
919    revisions: Box<[AtomicRevision]>,
920    _configuration: PhantomData<fn() -> C>,
921}
922
923// `#[salsa::interned(revisions = usize::MAX)]` disables garbage collection.
924const IMMORTAL: NonZeroUsize = NonZeroUsize::MAX;
925
926impl<C: Configuration> Default for RevisionQueue<C> {
927    fn default() -> RevisionQueue<C> {
928        let revisions = if C::REVISIONS == IMMORTAL {
929            Box::default()
930        } else {
931            (0..C::REVISIONS.get())
932                .map(|_| AtomicRevision::start())
933                .collect()
934        };
935
936        RevisionQueue {
937            lock: Mutex::new(()),
938            revisions,
939            _configuration: PhantomData,
940        }
941    }
942}
943
944impl<C: Configuration> RevisionQueue<C> {
945    /// Record the given revision as active.
946    #[inline]
947    fn record(&self, revision: Revision) {
948        // Garbage collection is disabled.
949        if C::REVISIONS == IMMORTAL {
950            return;
951        }
952
953        // Fast-path: We already recorded this revision.
954        if self.revisions[0].load() >= revision {
955            return;
956        }
957
958        self.record_cold(revision);
959    }
960
961    #[cold]
962    fn record_cold(&self, revision: Revision) {
963        let _lock = self.lock.lock();
964
965        // Otherwise, update the queue, maintaining sorted order.
966        //
967        // Note that this should only happen once per revision.
968        for i in (1..C::REVISIONS.get()).rev() {
969            self.revisions[i].store(self.revisions[i - 1].load());
970        }
971
972        self.revisions[0].store(revision);
973    }
974
975    /// Returns `true` if the given revision is old enough to be considered stale.
976    #[inline]
977    fn is_stale(&self, revision: Revision) -> bool {
978        // Garbage collection is disabled.
979        if C::REVISIONS == IMMORTAL {
980            return false;
981        }
982
983        let oldest = self.revisions[C::REVISIONS.get() - 1].load();
984
985        // If we have not recorded `REVS` revisions yet, nothing can be stale.
986        if oldest == Revision::start() {
987            return false;
988        }
989
990        revision < oldest
991    }
992
993    /// Returns `true` if `C::REVISIONS` revisions have been recorded as active,
994    /// i.e. enough data has been recorded to start garbage collection.
995    #[inline]
996    fn is_primed(&self) -> bool {
997        // Garbage collection is disabled.
998        if C::REVISIONS == IMMORTAL {
999            return false;
1000        }
1001
1002        self.revisions[C::REVISIONS.get() - 1].load() > Revision::start()
1003    }
1004}

1006/// A trait for types that hash and compare like `O`.
1007pub trait HashEqLike<O> {
1008    fn hash<H: Hasher>(&self, h: &mut H);
1009    fn eq(&self, data: &O) -> bool;
1010}

1012/// The `Lookup` trait is a more flexible variant on [`std::borrow::Borrow`]
1013/// and [`std::borrow::ToOwned`].
1014///
1015/// It is implemented by "some type that can be used as the lookup key for `O`".
1016/// This means that `self` can be hashed and compared for equality with values
1017/// of type `O` without actually creating an owned value. It `self` needs to be interned,
1018/// it can be converted into an equivalent value of type `O`.
1019///
1020/// The canonical example is `&str: Lookup<String>`. However, this example
1021/// alone can be handled by [`std::borrow::Borrow`][]. In our case, we may have
1022/// multiple keys accumulated into a struct, like `ViewStruct: Lookup<(K1, ...)>`,
1023/// where `struct ViewStruct<L1: Lookup<K1>...>(K1...)`. The `Borrow` trait
1024/// requires that `&(K1...)` be convertible to `&ViewStruct` which just isn't
1025/// possible. `Lookup` instead offers direct `hash` and `eq` methods.
1026pub trait Lookup<O> {
1027    fn into_owned(self) -> O;
1028}
1029
1030impl<T> Lookup<T> for T {
1031    fn into_owned(self) -> T {
1032        self
1033    }
1034}
