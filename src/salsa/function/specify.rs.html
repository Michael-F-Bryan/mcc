11impl<C> IngredientImpl<C>
12where
13    C: Configuration,
14{
15    /// Specify the value for `key` *and* record that we did so.
16    /// Used for explicit calls to `specify`, but not needed for pre-declared tracked struct fields.
17    pub fn specify_and_record<'db>(&'db self, db: &'db C::DbView, key: Id, value: C::Output<'db>)
18    where
19        C::Input<'db>: TrackedStructInDb,
20    {
21        let (zalsa, zalsa_local) = db.zalsas();
22
23        let (active_query_key, current_deps) = match zalsa_local.active_query() {
24            Some(v) => v,
25            None => panic!("can only use `specify` inside a tracked function"),
26        };
27
28        // `specify` only works if the key is a tracked struct created in the current query.
29        //
30        // The reason is this. We want to ensure that the same result is reached regardless of
31        // the "path" that the user takes through the execution graph.
32        // If you permit values to be specified from other queries, you can have a situation like this:
33        // * Q0 creates the tracked struct T0
34        // * Q1 specifies the value for F(T0)
35        // * Q2 invokes F(T0)
36        // * Q3 invokes Q1 and then Q2
37        // * Q4 invokes Q2 and then Q1
38        //
39        // Now, if We invoke Q3 first, We get one result for Q2, but if We invoke Q4 first, We get a different value. That's no good.
40        let database_key_index = <C::Input<'db>>::database_key_index(zalsa, key);
41        if !zalsa_local.is_output_of_active_query(database_key_index) {
42            panic!("can only use `specify` on salsa structs created during the current tracked fn");
43        }
44
45        // Subtle: we treat the "input" to a set query as if it were
46        // volatile.
47        //
48        // The idea is this. You have the current query C that
49        // created the entity E, and it is setting the value F(E) of the function F.
50        // When some other query R reads the field F(E), in order to have obtained
51        // the entity E, it has to have executed the query C.
52        //
53        // This will have forced C to either:
54        //
55        // - not create E this time, in which case R shouldn't have it (some kind of leak has occurred)
56        // - assign a value to F(E), in which case `verified_at` will be the current revision and `changed_at` will be updated appropriately
57        // - NOT assign a value to F(E), in which case we need to re-execute the function (which typically panics).
58        //
59        // So, ruling out the case of a leak having occurred, that means that the reader R will either see:
60        //
61        // - a result that is verified in the current revision, because it was set, which will use the set value
62        // - a result that is NOT verified and has untracked inputs, which will re-execute (and likely panic)
63
64        let revision = zalsa.current_revision();
65        let mut revisions = QueryRevisions {
66            changed_at: current_deps.changed_at,
67            durability: current_deps.durability,
68            origin: QueryOrigin::assigned(active_query_key),
69            accumulated_inputs: Default::default(),
70            verified_final: AtomicBool::new(true),
71            extra: QueryRevisionsExtra::default(),
72        };
73
74        let memo_ingredient_index = self.memo_ingredient_index(zalsa, key);
75        if let Some(old_memo) = self.get_memo_from_table_for(zalsa, key, memo_ingredient_index) {
76            self.backdate_if_appropriate(old_memo, database_key_index, &mut revisions, &value);
77            self.diff_outputs(zalsa, database_key_index, old_memo, &mut revisions);
78        }
79
80        let memo = Memo {
81            value: Some(value),
82            verified_at: AtomicRevision::from(revision),
83            revisions,
84        };
85
86        tracing::debug!(
87            "specify: about to add memo {:#?} for key {:?}",
88            memo.tracing_debug(),
89            key
90        );
91        self.insert_memo(zalsa, key, memo, memo_ingredient_index);
92
93        // Record that the current query *specified* a value for this cell.
94        let database_key_index = self.database_key_index(key);
95        zalsa_local.add_output(database_key_index);
96    }
97
98    /// Invoked when the query `executor` has been validated as having green inputs
99    /// and `key` is a value that was specified by `executor`.
100    /// Marks `key` as valid in the current revision since if `executor` had re-executed,
101    /// it would have specified `key` again.
102    pub(super) fn validate_specified_value(
103        &self,
104        zalsa: &Zalsa,
105        executor: DatabaseKeyIndex,
106        key: Id,
107    ) {
108        let memo_ingredient_index = self.memo_ingredient_index(zalsa, key);
109
110        let memo = match self.get_memo_from_table_for(zalsa, key, memo_ingredient_index) {
111            Some(m) => m,
112            None => return,
113        };
114
115        // If we are marking this as validated, it must be a value that was
116        // assigned by `executor`.
117        match memo.revisions.origin.as_ref() {
118            QueryOriginRef::Assigned(by_query) => assert_eq!(by_query, executor),
119            _ => panic!(
120                "expected a query assigned by `{:?}`, not `{:?}`",
121                executor, memo.revisions.origin,
122            ),
123        }
124
125        let database_key_index = self.database_key_index(key);
126        memo.mark_as_verified(zalsa, database_key_index);
127        memo.revisions
128            .accumulated_inputs
129            .store(InputAccumulatedValues::Empty);
130    }
