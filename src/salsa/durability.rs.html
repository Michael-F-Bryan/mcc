1/// Describes how likely a value is to changeâ€”how "durable" it is.
2///
3/// By default, inputs have `Durability::LOW` and interned values have
4/// `Durability::HIGH`. But inputs can be explicitly set with other
5/// durabilities.
6///
7/// We use durabilities to optimize the work of "revalidating" a query
8/// after some input has changed. Ordinarily, in a new revision,
9/// queries have to trace all their inputs back to the base inputs to
10/// determine if any of those inputs have changed. But if we know that
11/// the only changes were to inputs of low durability (the common
12/// case), and we know that the query only used inputs of medium
13/// durability or higher, then we can skip that enumeration.
14///
15/// Typically, one assigns low durabilities to inputs that the user is
16/// frequently editing. Medium or high durabilities are used for
17/// configuration, the source from library crates, or other things
18/// that are unlikely to be edited.
19#[derive(Copy, Clone, PartialEq, Eq, PartialOrd, Ord)]
20pub struct Durability(DurabilityVal);
21
22impl std::fmt::Debug for Durability {
23    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
24        if f.alternate() {
25            match self.0 {
26                DurabilityVal::Low => f.write_str("Durability::LOW"),
27                DurabilityVal::Medium => f.write_str("Durability::MEDIUM"),
28                DurabilityVal::High => f.write_str("Durability::HIGH"),
29            }
30        } else {
31            f.debug_tuple("Durability")
32                .field(&(self.0 as usize))
33                .finish()
34        }
35    }
36}
37
38// We use an enum here instead of a u8 for niches.
39#[derive(Copy, Clone, PartialEq, Eq, PartialOrd, Ord)]
40enum DurabilityVal {
41    Low = 0,
42    Medium = 1,
43    High = 2,
44}
45
46impl From<u8> for DurabilityVal {
47    fn from(value: u8) -> Self {
48        match value {
49            0 => DurabilityVal::Low,
50            1 => DurabilityVal::Medium,
51            2 => DurabilityVal::High,
52            _ => panic!("invalid durability"),
53        }
54    }
55}
56
57impl Durability {
58    /// Low durability: things that change frequently.
59    ///
60    /// Example: part of the crate being edited
61    pub const LOW: Durability = Durability(DurabilityVal::Low);
62
63    /// Medium durability: things that change sometimes, but rarely.
64    ///
65    /// Example: a Cargo.toml file
66    pub const MEDIUM: Durability = Durability(DurabilityVal::Medium);
67
68    /// High durability: things that are not expected to change under
69    /// common usage.
70    ///
71    /// Example: the standard library or something from crates.io
72    pub const HIGH: Durability = Durability(DurabilityVal::High);
73
74    /// The minimum possible durability; equivalent to LOW but
75    /// "conceptually" distinct (i.e., if we add more durability
76    /// levels, this could change).
77    pub(crate) const MIN: Durability = Self::LOW;
78
79    /// The maximum possible durability; equivalent to HIGH but
80    /// "conceptually" distinct (i.e., if we add more durability
81    /// levels, this could change).
82    pub(crate) const MAX: Durability = Self::HIGH;
83
84    /// Number of durability levels.
85    pub(crate) const LEN: usize = Self::HIGH.0 as usize + 1;
86
87    pub(crate) fn index(self) -> usize {
88        self.0 as usize
89    }
90}
91
92impl Default for Durability {
93    fn default() -> Self {
94        Durability::LOW
95    }
