21/// State that is specific to a single execution thread.
22///
23/// Internally, this type uses ref-cells.
24///
25/// **Note also that all mutations to the database handle (and hence
26/// to the local-state) must be undone during unwinding.**
27pub struct ZalsaLocal {
28    /// Vector of active queries.
29    ///
30    /// Unwinding note: pushes onto this vector must be popped -- even
31    /// during unwinding.
32    query_stack: RefCell<QueryStack>,

35    /// This is thread-local to avoid contention.

37}

47    pub(crate) fn record_unfilled_pages(&mut self, table: &Table) {
48        let most_recent_pages = self.most_recent_pages.get_mut();
49        most_recent_pages
50            .drain()
51            .for_each(|(ingredient, page)| table.record_unfilled_page(ingredient, page));
52    }

54    /// Allocate a new id in `table` for the given ingredient
55    /// storing `value`. Remembers the most recent page from this
56    /// thread and attempts to reuse it.
57    pub(crate) fn allocate<T: Slot>(
58        &self,
59        zalsa: &Zalsa,
60        ingredient: IngredientIndex,
61        mut value: impl FnOnce(Id) -> T,
62    ) -> Id {
63        let memo_types = || {
64            zalsa
65                .lookup_ingredient(ingredient)
66                .memo_table_types()
67                .clone()
68        };
69        // Find the most recent page, pushing a page if needed
70        let mut page = *self
71            .most_recent_pages
72            .borrow_mut()
73            .entry(ingredient)
74            .or_insert_with(|| {
75                zalsa
76                    .table()
77                    .fetch_or_push_page::<T>(ingredient, memo_types)
78            });
79
80        loop {
81            // Try to allocate an entry on that page
82            let page_ref = zalsa.table().page::<T>(page);
83            match page_ref.allocate(page, value) {
84                // If successful, return
85                Ok(id) => return id,
86
87                // Otherwise, create a new page and try again
88                // Note that we could try fetching a page again, but as we just filled one up
89                // it is unlikely that there is a non-full one available.
90                Err(v) => {
91                    value = v;
92                    page = zalsa.table().push_page::<T>(ingredient, memo_types());
93                    self.most_recent_pages.borrow_mut().insert(ingredient, page);
94                }
95            }
96        }
97    }

263    /// Called when the active queries creates an index from the
264    /// entity table with the index `entity_index`. Has the following effects:
265    ///
266    /// * Add a query read on `DatabaseKeyIndex::for_table(entity_index)`
267    /// * Identify a unique disambiguator for the hash within the current query,
268    ///   adding the hash to the current query's disambiguator table.
269    /// * Returns a tuple of:
270    ///   * the id of the current query
271    ///   * the current dependencies (durability, changed_at) of current query
272    ///   * the disambiguator index
273    #[track_caller]
274    pub(crate) fn disambiguate(&self, key: IdentityHash) -> (Stamp, Disambiguator) {
275        self.with_query_stack_mut(|stack| {
276            let top_query = stack.last_mut().expect(
277                "cannot create a tracked struct disambiguator outside of a tracked function",
278            );
279            let disambiguator = top_query.disambiguate(key);
280            (top_query.stamp(), disambiguator)
281        })
282    }

318/// Summarizes "all the inputs that a query used"
319/// and "all the outputs it has written to"

351    #[cfg(feature = "salsa_unstable")]
352    pub(crate) fn allocation_size(&self) -> usize {
353        let QueryRevisions {
354            changed_at: _,
355            durability: _,
356            accumulated_inputs: _,
357            verified_final: _,
358            origin,
359            extra,
360        } = self;
361
362        let mut memory = 0;
363
364        if let QueryOriginRef::Derived(query_edges)
365        | QueryOriginRef::DerivedUntracked(query_edges) = origin.as_ref()
366        {
367            memory += std::mem::size_of_val(query_edges);
368        }
369
370        if let Some(extra) = extra.0.as_ref() {
371            memory += std::mem::size_of::<QueryRevisionsExtra>();
372            memory += extra.allocation_size();
373        }
374
375        memory
376    }
377}

379/// Data on `QueryRevisions` that is lazily allocated to save memory
380/// in the common case.
381///
382/// In particular, not all queries create tracked structs, participate
383/// in cycles, or create accumulators.
384#[derive(Debug, Default)]
385pub(crate) struct QueryRevisionsExtra(Option<Box<QueryRevisionsExtraInner>>);
386
387impl QueryRevisionsExtra {
388    pub fn new(
389        accumulated: AccumulatedMap,
390        tracked_struct_ids: IdentityMap,
391        cycle_heads: CycleHeads,
392        iteration: IterationCount,
393    ) -> Self {
394        let inner = if tracked_struct_ids.is_empty()
395            && cycle_heads.is_empty()
396            && accumulated.is_empty()
397            && iteration.is_initial()
398        {
399            None
400        } else {
401            Some(Box::new(QueryRevisionsExtraInner {
402                accumulated,
403                cycle_heads,
404                tracked_struct_ids: tracked_struct_ids.into_thin_vec(),
405                iteration,
406            }))
407        };
408
409        Self(inner)
410    }
411}

413#[derive(Debug)]
414struct QueryRevisionsExtraInner {
415    accumulated: AccumulatedMap,
416
417    /// The ids of tracked structs created by this query.
418    ///
419    /// This table plays an important role when queries are
420    /// re-executed:
421    /// * A clone of this field is used as the initial set of
422    ///   `TrackedStructId`s for the query on the next execution.
423    /// * The query will thus re-use the same ids if it creates
424    ///   tracked structs with the same `KeyStruct` as before.
425    ///   It may also create new tracked structs.
426    /// * One tricky case involves deleted structs. If
427    ///   the old revision created a struct S but the new
428    ///   revision did not, there will still be a map entry
429    ///   for S. This is because queries only ever grow the map
430    ///   and they start with the same entries as from the
431    ///   previous revision. To handle this, `diff_outputs` compares
432    ///   the structs from the old/new revision and retains
433    ///   only entries that appeared in the new revision.
434    tracked_struct_ids: ThinVec<(Identity, Id)>,
435
436    /// This result was computed based on provisional values from
437    /// these cycle heads. The "cycle head" is the query responsible
438    /// for managing a fixpoint iteration. In a cycle like
439    /// `--> A --> B --> C --> A`, the cycle head is query `A`: it is
440    /// the query whose value is requested while it is executing,
441    /// which must provide the initial provisional value and decide,
442    /// after each iteration, whether the cycle has converged or must
443    /// iterate again.
444    cycle_heads: CycleHeads,
445
446    iteration: IterationCount,
447}

449impl QueryRevisionsExtraInner {
450    #[cfg(feature = "salsa_unstable")]
451    fn allocation_size(&self) -> usize {
452        let QueryRevisionsExtraInner {
453            accumulated,
454            tracked_struct_ids,
455            cycle_heads,
456            iteration: _,
457        } = self;
458
459        accumulated.allocation_size()
460            + cycle_heads.allocation_size()
461            + std::mem::size_of_val(tracked_struct_ids.as_slice())
462    }
463}

565/// Tracks the way that a memoized value for a query was created.
566///
567/// This is a read-only reference to a `PackedQueryOrigin`.
568#[derive(Debug, Clone, Copy)]
569#[repr(u8)]
570pub enum QueryOriginRef<'a> {
571    /// The value was assigned as the output of another query (e.g., using `specify`).
572    /// The `DatabaseKeyIndex` is the identity of the assigning query.
573    Assigned(DatabaseKeyIndex) = QueryOriginKind::Assigned as u8,
574
575    /// The value was derived by executing a function
576    /// and we were able to track ALL of that function's inputs.
577    /// Those inputs are described in [`QueryEdges`].
578    Derived(&'a [QueryEdge]) = QueryOriginKind::Derived as u8,
579
580    /// The value was derived by executing a function
581    /// but that function also reported that it read untracked inputs.
582    /// The [`QueryEdges`] argument contains a listing of all the inputs we saw
583    /// (but we know there were more).
584    DerivedUntracked(&'a [QueryEdge]) = QueryOriginKind::DerivedUntracked as u8,
585
586    /// The value is an initial provisional value for a query that supports fixpoint iteration.
587    FixpointInitial = QueryOriginKind::FixpointInitial as u8,
588}
589
590impl<'a> QueryOriginRef<'a> {
591    /// Indices for queries *read* by this query
592    #[inline]
593    pub(crate) fn inputs(self) -> impl DoubleEndedIterator<Item = DatabaseKeyIndex> + use<'a> {
594        let opt_edges = match self {
595            QueryOriginRef::Derived(edges) | QueryOriginRef::DerivedUntracked(edges) => Some(edges),
596            QueryOriginRef::Assigned(_) | QueryOriginRef::FixpointInitial => None,
597        };
598        opt_edges.into_iter().flat_map(input_edges)
599    }
600
601    /// Indices for queries *written* by this query (if any)
602    pub(crate) fn outputs(self) -> impl DoubleEndedIterator<Item = DatabaseKeyIndex> + use<'a> {
603        let opt_edges = match self {
604            QueryOriginRef::Derived(edges) | QueryOriginRef::DerivedUntracked(edges) => Some(edges),
605            QueryOriginRef::Assigned(_) | QueryOriginRef::FixpointInitial => None,
606        };
607        opt_edges.into_iter().flat_map(output_edges)
608    }
609
610    #[inline]
611    pub(crate) fn edges(self) -> &'a [QueryEdge] {
612        let opt_edges = match self {
613            QueryOriginRef::Derived(edges) | QueryOriginRef::DerivedUntracked(edges) => Some(edges),
614            QueryOriginRef::Assigned(_) | QueryOriginRef::FixpointInitial => None,
615        };
616
617        opt_edges.unwrap_or_default()
618    }
619}

621// Note: The discriminant assignment is intentional,
622// we want to group `Derived` and `DerivedUntracked` together on a same bit (the second LSB)
623// as we tend to match against both of them in the same branch.
624#[derive(Clone, Copy)]
625#[repr(u8)]
626enum QueryOriginKind {
627    /// An initial provisional value.
628    ///
629    /// This will occur occur in queries that support fixpoint iteration.
630    FixpointInitial = 0b00,
631
632    /// The value was assigned as the output of another query.
633    ///
634    /// This can, for example, can occur when `specify` is used.
635    Assigned = 0b01,
636
637    /// The value was derived by executing a function
638    /// _and_ Salsa was able to track all of said function's inputs.
639    Derived = 0b11,
640
641    /// The value was derived by executing a function
642    /// but that function also reported that it read untracked inputs.
643    DerivedUntracked = 0b10,
644}

646/// Tracks how a memoized value for a given query was created.
647///
648/// This type is a manual enum packed to 13 bytes to reduce the size of `QueryRevisions`.
649#[repr(Rust, packed)]
650pub struct QueryOrigin {
651    /// The tag of this enum.
652    ///
653    /// Note that this tag only requires two bits and could likely be packed into
654    /// some other field. However, we get this byte for free thanks to alignment.
655    kind: QueryOriginKind,
656
657    /// The data portion of this enum.
658    data: QueryOriginData,
659
660    /// The metadata of this enum.
661    ///
662    /// For `QueryOriginKind::Derived` and `QueryOriginKind::DerivedUntracked`, this
663    /// is the length of the `input_outputs` allocation.
664    ///
665    /// For `QueryOriginKind::Assigned`, this is the `IngredientIndex` of assigning query.
666    /// Combined with the `Id` data, this forms a complete `DatabaseKeyIndex`.
667    ///
668    /// For `QueryOriginKind::FixpointInitial`, this field is zero.
669    metadata: u32,
670}
671
672/// The data portion of `PackedQueryOrigin`.
673union QueryOriginData {
674    /// Query edges for `QueryOriginKind::Derived` or `QueryOriginKind::DerivedUntracked`.
675    ///
676    /// The query edges are between a memoized value and other queries in the dependency graph,
677    /// including both dependency edges (e.g., when creating the memoized value for Q0
678    /// executed another function Q1) and output edges (e.g., when Q0 specified the value
679    /// for another query Q2).
680    ///
681    /// Note that we always track input dependencies even when there are untracked reads.
682    /// Untracked reads mean that Salsa can't verify values, so the list of inputs is unused.
683    /// However, Salsa still uses these edges to find the transitive inputs to an accumulator.
684    ///
685    /// You can access the input/output list via the methods [`inputs`] and [`outputs`] respectively.
686    ///
687    /// Important:
688    ///
689    /// * The inputs must be in **execution order** for the red-green algorithm to work.
690    input_outputs: NonNull<QueryEdge>,
691
692    /// The identity of the assigning query for `QueryOriginKind::Assigned`.
693    index: Id,
694
695    /// `QueryOriginKind::FixpointInitial` holds no data.
696    empty: (),
697}
698
699/// SAFETY: The `input_outputs` pointer is owned and not accessed or shared concurrently.
700unsafe impl Send for QueryOriginData {}
701/// SAFETY: Same as above.
702unsafe impl Sync for QueryOriginData {}
703
704impl QueryOrigin {
705    /// Create a query origin of type `QueryOriginKind::FixpointInitial`.
706    pub fn fixpoint_initial() -> QueryOrigin {
707        QueryOrigin {
708            kind: QueryOriginKind::FixpointInitial,
709            metadata: 0,
710            data: QueryOriginData { empty: () },
711        }
712    }
713
714    /// Create a query origin of type `QueryOriginKind::Derived`, with the given edges.
715    pub fn derived(input_outputs: impl IntoIterator<Item = QueryEdge>) -> QueryOrigin {
716        let input_outputs = input_outputs.into_iter().collect::<Box<[_]>>();
717
718        // Exceeding `u32::MAX` query edges should never happen in real-world usage.
719        let length = u32::try_from(input_outputs.len())
720            .expect("exceeded more than `u32::MAX` query edges; this should never happen.");
721
722        // SAFETY: `Box::into_raw` returns a non-null pointer.
723        let input_outputs =
724            unsafe { NonNull::new_unchecked(Box::into_raw(input_outputs).cast::<QueryEdge>()) };
725
726        QueryOrigin {
727            kind: QueryOriginKind::Derived,
728            metadata: length,
729            data: QueryOriginData { input_outputs },
730        }
731    }
732
733    /// Create a query origin of type `QueryOriginKind::DerivedUntracked`, with the given edges.
734    pub fn derived_untracked(input_outputs: impl IntoIterator<Item = QueryEdge>) -> QueryOrigin {
735        let mut origin = QueryOrigin::derived(input_outputs);
736        origin.kind = QueryOriginKind::DerivedUntracked;
737        origin
738    }
739
740    /// Create a query origin of type `QueryOriginKind::Assigned`, with the given key.
741    pub fn assigned(key: DatabaseKeyIndex) -> QueryOrigin {
742        QueryOrigin {
743            kind: QueryOriginKind::Assigned,
744            metadata: key.ingredient_index().as_u32(),
745            data: QueryOriginData {
746                index: key.key_index(),
747            },
748        }
749    }
750
751    /// Return a read-only reference to this query origin.
752    pub fn as_ref(&self) -> QueryOriginRef<'_> {
753        match self.kind {
754            QueryOriginKind::Assigned => {
755                // SAFETY: `data.index` is initialized when the tag is `QueryOriginKind::Assigned`.
756                let index = unsafe { self.data.index };
757                let ingredient_index = IngredientIndex::from(self.metadata as usize);
758                QueryOriginRef::Assigned(DatabaseKeyIndex::new(ingredient_index, index))
759            }
760
761            QueryOriginKind::Derived => {
762                // SAFETY: `data.input_outputs` is initialized when the tag is `QueryOriginKind::Derived`.
763                let input_outputs = unsafe { self.data.input_outputs };
764                let length = self.metadata as usize;
765
766                // SAFETY: `input_outputs` and `self.metadata` form a valid slice when the
767                // tag is `QueryOriginKind::Derived`.
768                let input_outputs =
769                    unsafe { std::slice::from_raw_parts(input_outputs.as_ptr(), length) };
770
771                QueryOriginRef::Derived(input_outputs)
772            }
773
774            QueryOriginKind::DerivedUntracked => {
775                // SAFETY: `data.input_outputs` is initialized when the tag is `QueryOriginKind::DerivedUntracked`.
776                let input_outputs = unsafe { self.data.input_outputs };
777                let length = self.metadata as usize;
778
779                // SAFETY: `input_outputs` and `self.metadata` form a valid slice when the
780                // tag is `QueryOriginKind::DerivedUntracked`.
781                let input_outputs =
782                    unsafe { std::slice::from_raw_parts(input_outputs.as_ptr(), length) };
783
784                QueryOriginRef::DerivedUntracked(input_outputs)
785            }
786
787            QueryOriginKind::FixpointInitial => QueryOriginRef::FixpointInitial,
788        }
789    }
790}
791
792impl Drop for QueryOrigin {
793    fn drop(&mut self) {
794        match self.kind {
795            QueryOriginKind::Derived | QueryOriginKind::DerivedUntracked => {
796                // SAFETY: `data.input_outputs` is initialized when the tag is `QueryOriginKind::Derived`
797                // or `QueryOriginKind::DerivedUntracked`.
798                let input_outputs = unsafe { self.data.input_outputs };
799                let length = self.metadata as usize;
800
801                // SAFETY: `input_outputs` and `self.metadata` form a valid slice when the
802                // tag is `QueryOriginKind::DerivedUntracked` or `QueryOriginKind::DerivedUntracked`,
803                // and we have `&mut self`.
804                let _input_outputs: Box<[QueryEdge]> = unsafe {
805                    Box::from_raw(ptr::slice_from_raw_parts_mut(
806                        input_outputs.as_ptr(),
807                        length,
808                    ))
809                };
810            }
811
812            // The data stored for this variants is `Copy`.
813            QueryOriginKind::FixpointInitial | QueryOriginKind::Assigned => {}
814        }
815    }
816}

818impl std::fmt::Debug for QueryOrigin {
819    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
820        self.as_ref().fmt(f)
821    }
822}

824/// An input or output query edge.
825///
826/// This type is a packed version of `QueryEdgeKind`, tagging the `IngredientIndex`
827/// in `key` with a discriminator for the input and output variants without increasing
828/// the size of the type. Notably, this type is 12 bytes as opposed to the 16 byte
829/// `QueryEdgeKind`, which is meaningful as inputs and outputs are stored contiguously.
830#[derive(Copy, Clone, PartialEq, Eq, Hash)]
831pub struct QueryEdge {
832    key: DatabaseKeyIndex,
833}
834
835impl QueryEdge {
836    /// Create an input query edge with the given index.
837    pub fn input(key: DatabaseKeyIndex) -> QueryEdge {
838        Self { key }
839    }
840
841    /// Create an output query edge with the given index.
842    pub fn output(key: DatabaseKeyIndex) -> QueryEdge {
843        let ingredient_index = key.ingredient_index().with_tag(true);
844
845        Self {
846            key: DatabaseKeyIndex::new(ingredient_index, key.key_index()),
847        }
848    }
849
850    /// Returns the kind of this query edge.
851    pub fn kind(self) -> QueryEdgeKind {
852        // Clear the tag to restore the original index.
853        let untagged = DatabaseKeyIndex::new(
854            self.key.ingredient_index().with_tag(false),
855            self.key.key_index(),
856        );
857
858        if self.key.ingredient_index().tag() {
859            QueryEdgeKind::Output(untagged)
860        } else {
861            QueryEdgeKind::Input(untagged)
862        }
863    }
864}
865
866impl std::fmt::Debug for QueryEdge {
867    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
868        self.kind().fmt(f)
869    }
870}
871
872#[derive(Copy, Clone, Debug, PartialEq, Eq, Hash)]
873pub enum QueryEdgeKind {
874    Input(DatabaseKeyIndex),
875    Output(DatabaseKeyIndex),
876}
877
878/// Returns the (tracked) inputs that were executed in computing this memoized value.
879///
880/// These will always be in execution order.
881pub(crate) fn input_edges(
882    input_outputs: &[QueryEdge],
883) -> impl DoubleEndedIterator<Item = DatabaseKeyIndex> + use<'_> {
884    input_outputs.iter().filter_map(|&edge| match edge.kind() {
885        QueryEdgeKind::Input(dependency_index) => Some(dependency_index),
886        QueryEdgeKind::Output(_) => None,
887    })
888}
889
890/// Returns the (tracked) outputs that were executed in computing this memoized value.
891///
892/// These will always be in execution order.
893pub(crate) fn output_edges(
894    input_outputs: &[QueryEdge],
895) -> impl DoubleEndedIterator<Item = DatabaseKeyIndex> + use<'_> {
896    input_outputs.iter().filter_map(|&edge| match edge.kind() {
897        QueryEdgeKind::Output(dependency_index) => Some(dependency_index),
898        QueryEdgeKind::Input(_) => None,
899    })
900}

902/// When a query is pushed onto the `active_query` stack, this guard
903/// is returned to represent its slot. The guard can be used to pop
904/// the query from the stack -- in the case of unwinding, the guard's
905/// destructor will also remove the query.
906pub(crate) struct ActiveQueryGuard<'me> {
907    local_state: &'me ZalsaLocal,
908    #[cfg(debug_assertions)]
909    push_len: usize,
910    pub(crate) database_key_index: DatabaseKeyIndex,
911}
912
913impl ActiveQueryGuard<'_> {
914    /// Initialize the tracked struct ids with the values from the prior execution.
915    pub(crate) fn seed_tracked_struct_ids(&self, tracked_struct_ids: &[(Identity, Id)]) {
916        self.local_state.with_query_stack_mut(|stack| {
917            #[cfg(debug_assertions)]
918            assert_eq!(stack.len(), self.push_len);
919            let frame = stack.last_mut().unwrap();
920            assert!(frame.tracked_struct_ids().is_empty());
921            frame
922                .tracked_struct_ids_mut()
923                .clone_from_slice(tracked_struct_ids);
924        })
925    }
926
927    /// Append the given `outputs` to the query's output list.
928    pub(crate) fn seed_iteration(&self, previous: &QueryRevisions) {
929        let durability = previous.durability;
930        let changed_at = previous.changed_at;
931        let edges = previous.origin.as_ref().edges();
932        let untracked_read = matches!(
933            previous.origin.as_ref(),
934            QueryOriginRef::DerivedUntracked(_)
935        );
936
937        self.local_state.with_query_stack_mut(|stack| {
938            #[cfg(debug_assertions)]
939            assert_eq!(stack.len(), self.push_len);
940            let frame = stack.last_mut().unwrap();
941            frame.seed_iteration(durability, changed_at, edges, untracked_read);
942        })
943    }
944
945    /// Invoked when the query has successfully completed execution.
946    fn complete(self) -> QueryRevisions {
947        let query = self.local_state.with_query_stack_mut(|stack| {
948            stack.pop_into_revisions(
949                self.database_key_index,
950                #[cfg(debug_assertions)]
951                self.push_len,
952            )
953        });
954        std::mem::forget(self);
955        query
956    }
957
958    /// Pops an active query from the stack. Returns the [`QueryRevisions`]
959    /// which summarizes the other queries that were accessed during this
960    /// query's execution.
961    #[inline]
962    pub(crate) fn pop(self) -> QueryRevisions {
963        self.complete()
964    }
965}
966
967impl Drop for ActiveQueryGuard<'_> {
968    fn drop(&mut self) {
969        self.local_state.with_query_stack_mut(|stack| {
970            stack.pop(
971                self.database_key_index,
972                #[cfg(debug_assertions)]
973                self.push_len,
974            );
975        });
976    }
977}
