3use crate::memo_ingredient_indices::{IngredientIndices, MemoIngredientMap};
4use crate::zalsa::Zalsa;
5use crate::Id;

7pub trait SalsaStructInDb: Sized {
8    type MemoIngredientMap: MemoIngredientMap;
9
10    /// Lookup or create ingredient indices.
11    ///
12    /// Note that this method does *not* create the ingredients themselves, this is handled by
13    /// [`crate::zalsa::JarEntry::get_or_create`]. This method only creates
14    /// or looks up the indices corresponding to the ingredients.
15    ///
16    /// While implementors of this trait may call [`crate::zalsa::JarEntry::get_or_create`]
17    /// to create the ingredient, they aren't required to. For example, supertypes recursively
18    /// call [`crate::zalsa::JarEntry::get_or_create`] for their variants and combine them.
19    fn lookup_or_create_ingredient_index(zalsa: &Zalsa) -> IngredientIndices;
20
21    /// Plumbing to support nested salsa supertypes.
22    ///
23    /// In the example below, there are two supertypes: `InnerEnum` and `OuterEnum`,
24    /// where the former is a supertype of `Input` and `Interned1` and the latter
25    /// is a supertype of `InnerEnum` and `Interned2`.
26    ///
27    /// ```ignore
28    /// #[salsa::input]
29    /// struct Input {}
30    ///
31    /// #[salsa::interned]
32    /// struct Interned1 {}
33    ///
34    /// #[salsa::interned]
35    /// struct Interned2 {}
36    ///
37    /// #[derive(Debug, salsa::Enum)]
38    /// enum InnerEnum {
39    ///     Input(Input),
40    ///     Interned1(Interned1),
41    /// }
42    ///
43    /// #[derive(Debug, salsa::Enum)]
44    /// enum OuterEnum {
45    ///     InnerEnum(InnerEnum),
46    ///     Interned2(Interned2),
47    /// }
48    /// ```
49    ///
50    /// Imagine `OuterEnum` got a [`salsa::Id`][Id] and it wants to know which variant it belongs to.
51    ///
52    /// `OuterEnum` cannot ask each variant "what is your ingredient index?" and compare because `InnerEnum`
53    /// has *multiple*, possible ingredient indices. Alternatively, `OuterEnum` could ask eaach variant
54    /// "is this value yours?" and then invoke [`FromId`][crate::id::FromId] with the correct variant,
55    /// but this duplicates work: now, `InnerEnum` will have to repeat this check-and-cast for *its*
56    /// variants.
57    ///
58    /// Instead, the implementor keeps track of the [`std::any::TypeId`] of the ID struct, and ask each
59    /// variant to "cast" to it. If it succeeds, `cast` returns that value; if not, we
60    /// go to the next variant.
61    ///
62    /// Why `TypeId` and not `IngredientIndex`? Because it's cheaper and easier: the `TypeId` is readily
63    /// available at compile time, while the `IngredientIndex` requires a runtime lookup.
64    fn cast(id: Id, type_id: TypeId) -> Option<Self>;
65}
