6/// An ingredient has an [ingredient index][IngredientIndex]. However, Salsa also supports
7/// enums of salsa structs (and other salsa enums), and those don't have a constant ingredient index,
8/// because they are not ingredients by themselves but rather composed of them. However, an enum can
9/// be viewed as a *set* of [`IngredientIndex`], where each instance of the enum can belong
10/// to one, potentially different, index. This is what this type represents: a set of
11/// `IngredientIndex`.
12#[derive(Clone)]
13pub struct IngredientIndices {
14    indices: Box<[IngredientIndex]>,
15}
16
17impl From<IngredientIndex> for IngredientIndices {
18    #[inline]
19    fn from(value: IngredientIndex) -> Self {
20        Self {
21            indices: Box::new([value]),
22        }
23    }
24}
25
26impl IngredientIndices {
27    #[inline]
28    pub fn empty() -> Self {
29        Self {
30            indices: Box::default(),
31        }
32    }
33
34    pub fn merge(iter: impl IntoIterator<Item = Self>) -> Self {
35        let mut indices = Vec::new();
36        for index in iter {
37            indices.extend(index.indices);
38        }
39        indices.sort_unstable();
40        indices.dedup();
41        Self {
42            indices: indices.into_boxed_slice(),
43        }
44    }
45}

60impl NewMemoIngredientIndices for MemoIngredientIndices {
61    /// # Safety
62    ///
63    /// The memo types must be correct.
64    unsafe fn create(
65        zalsa: &Zalsa,
66        struct_indices: IngredientIndices,
67        ingredient: IngredientIndex,
68        memo_type: MemoEntryType,
69        _intern_ingredient_memo_types: Option<Arc<MemoTableTypes>>,
70    ) -> Self {
71        debug_assert!(
72            _intern_ingredient_memo_types.is_none(),
73            "intern ingredient can only have a singleton memo ingredient"
74        );
75        let Some(&last) = struct_indices.indices.last() else {
76            unreachable!("Attempting to construct struct memo mapping for non tracked function?")
77        };
78        let mut indices = Vec::new();
79        indices.resize(
80            (last.as_u32() as usize) + 1,
81            MemoIngredientIndex::from_usize((u32::MAX - 1) as usize),
82        );
83        for &struct_ingredient in &struct_indices.indices {
84            let memo_types = zalsa
85                .lookup_ingredient(struct_ingredient)
86                .memo_table_types();
87
88            let mi = zalsa.next_memo_ingredient_index(struct_ingredient, ingredient);
89            memo_types.set(mi, &memo_type);
90
91            indices[struct_ingredient.as_u32() as usize] = mi;
92        }
93        MemoIngredientIndices {
94            indices: indices.into_boxed_slice(),
95        }
96    }
97}

99/// This type is to [`MemoIngredientIndex`] what [`IngredientIndices`] is to [`IngredientIndex`]:
100/// since enums can contain different ingredient indices, they can also have different memo indices,
101/// so we need to keep track of them.
102///
103/// This acts a map from [`IngredientIndex`] to [`MemoIngredientIndex`] but implemented
104/// via a slice for fast lookups, trading memory for speed. With these changes, lookups are `O(1)`
105/// instead of `O(n)`.
106///
107/// A database tends to have few ingredients (i), less function ingredients and even less
108/// function ingredients targeting `#[derive(Supertype)]` enums (e).
109/// While this is bounded as `O(i * e)` memory usage, the average case is significantly smaller: a
110/// function ingredient targeting enums only stores a slice whose length corresponds to the largest
111/// ingredient index's _value_. For example, if we have the ingredient indices `[2, 6, 17]`, then we
112/// will allocate a slice whose length is `17 + 1`.
113///
114/// Assuming a heavy example scenario of 1000 ingredients (500 of which are function ingredients, 100
115/// of which are enum targeting functions) this would come out to a maximum possibly memory usage of
116/// 4bytes * 1000 * 100 ~= 0.38MB which is negligible.
117pub struct MemoIngredientIndices {
118    indices: Box<[MemoIngredientIndex]>,
119}
120
121impl MemoIngredientMap for MemoIngredientIndices {
122    #[inline(always)]
123    fn get_zalsa_id(&self, zalsa: &Zalsa, id: Id) -> MemoIngredientIndex {
124        self.get(zalsa.ingredient_index(id))
125    }
126    #[inline(always)]
127    fn get(&self, index: IngredientIndex) -> MemoIngredientIndex {
128        self.indices[index.as_u32() as usize]
129    }
130}

146impl NewMemoIngredientIndices for MemoIngredientSingletonIndex {
147    #[inline]
148    unsafe fn create(
149        zalsa: &Zalsa,
150        indices: IngredientIndices,
151        ingredient: IngredientIndex,
152        memo_type: MemoEntryType,
153        intern_ingredient_memo_types: Option<Arc<MemoTableTypes>>,
154    ) -> Self {
155        let &[struct_ingredient] = &*indices.indices else {
156            unreachable!("Attempting to construct struct memo mapping from enum?")
157        };
158
159        let memo_types = intern_ingredient_memo_types.unwrap_or_else(|| {
160            zalsa
161                .lookup_ingredient(struct_ingredient)
162                .memo_table_types()
163        });
164
165        let mi = zalsa.next_memo_ingredient_index(struct_ingredient, ingredient);
166        memo_types.set(mi, &memo_type);
167        Self(mi)
168    }
169}
