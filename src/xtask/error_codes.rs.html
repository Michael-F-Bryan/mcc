14#[derive(Debug, Parser, Clone, PartialEq)]
15pub struct ErrorCodes {
16    #[clap(short, long, default_value = CODES_RS.as_os_str())]
17    output: PathBuf,
18    #[clap(default_value = ERROR_CODES_YAML.as_os_str())]
19    input: PathBuf,
20}

22impl ErrorCodes {
23    pub fn run(self) -> anyhow::Result<()> {
24        let ErrorCodes { output, input } = self;
25
26        let src = std::fs::read_to_string(&input)
27            .with_context(|| format!("reading \"{}\"", input.display()))?;
28        let root_namespace: BTreeMap<String, Value> = serde_yaml::from_str(&src)
29            .with_context(|| format!("parsing \"{}\"", input.display()))?;
30
31        let tokens = generate_codes_rs(root_namespace).to_token_stream();
32        crate::ensure_file_contents(&output, tokens, "error_codes");
33
34        Ok(())
35    }
36}

38fn generate_codes_rs(root_namespace: BTreeMap<String, Value>) -> impl ToTokens {
39    let mut segments = Vec::new();
40    let (tokens, error_codes) = generate_namespace(&mut segments, &root_namespace);
41
42    quote! {
43        //! Common error codes used across the compiler.
44        #![allow(non_upper_case_globals)]
45        use codespan_reporting::diagnostic::Severity;
46
47        #[derive(Debug, Clone, PartialEq, Eq, Ord, PartialOrd, Hash)]
48        pub struct ErrorCode {
49            pub segments: &'static [&'static str],
50            pub severity: Severity,
51            pub description: &'static str,
52        }
53
54        impl std::fmt::Display for ErrorCode {
55            fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
56                for (i, segment) in self.segments.iter().enumerate() {
57                    if i > 0 {
58                        write!(f, "::")?;
59                    }
60                    write!(f, "{}", segment)?;
61                }
62                Ok(())
63            }
64        }
65
66        /// All error codes.
67        pub const ALL: &[ErrorCode] = &[ #(#error_codes,)* ];
68
69        /// The error codes definition, as YAML.
70        pub const DEFINITION: &str = include_str!(
71            concat!(env!("CARGO_MANIFEST_DIR"), "/error-codes.yaml")
72        );
73
74        #tokens
75    }
76}

78fn generate_namespace<'a>(
79    segments: &mut Vec<&'a str>,
80    namespace: &'a BTreeMap<String, Value>,
81) -> (TokenStream, Vec<TokenStream>) {
82    let mut tokens = TokenStream::new();
83    let mut error_codes = Vec::new();
84
85    for (name, value) in namespace {
86        match value {
87            Value::ErrorCode(error_code) => {
88                error_codes.push(quote::format_ident!("{name}").to_token_stream());
89                tokens.extend(generate_error_code(segments, name, error_code));
90            }
91            Value::Namespace(namespace) => {
92                segments.push(name);
93                let (toks, child_codes) = generate_namespace(&mut *segments, namespace);
94                let ident = quote::format_ident!("{name}");
95                for child in child_codes {
96                    let new_code = quote!(#ident::#child);
97                    error_codes.push(new_code);
98                }
99                tokens.extend(quote! {
100                    pub mod #ident {
101                        use super::*;
102                        #toks
103                    }
104                });
105                segments.pop();
106            }
107        }
108    }
109
110    (tokens, error_codes)
111}

113fn generate_error_code(segments: &[&str], name: &str, error_code: &ErrorCode) -> TokenStream {
114    let ErrorCode {
115        severity,
116        description,
117    } = error_code;
118    let ident = quote::format_ident!("{name}");
119
120    quote::quote! {
121        #[doc = #description]
122        pub const #ident: ErrorCode = ErrorCode {
123            segments: &[#(#segments,)* #name],
124            severity: #severity,
125            description: #description,
126        };
127    }
128}

137#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
138struct ErrorCode {
139    severity: Severity,
140    description: String,
141}

143#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
144enum Severity {
145    Bug,
146    Error,
147    Warning,
148    Note,
149    Help,
150}
151
152impl ToTokens for Severity {
153    fn to_tokens(&self, tokens: &mut TokenStream) {
154        let name = match self {
155            Severity::Bug => "Bug",
156            Severity::Error => "Error",
157            Severity::Warning => "Warning",
158            Severity::Note => "Note",
159            Severity::Help => "Help",
160        };
161        let variant = quote::format_ident!("{name}");
162        tokens.extend(quote!(Severity::#variant));
163    }
164}
